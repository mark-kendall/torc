64 bool TorcIOTLogger::Initialise(const QVariantMap &Details)
65 {
66  if (Details.contains("apikey"))
67  m_apiKey = Details.value("apikey").toString().trimmed();
68 
69  if (m_apiKey.isEmpty())
70  {
71  LOG(VB_GENERAL, LOG_ERR, QString("%1 logger has no api key - disabling").arg(m_description));
72  return false;
73  }
74 
75  if (Details.contains("fields"))
76  {
77  QVariantMap fields = Details.value("fields").toMap();
78  for (int i = 0; i < m_maxFields; i++)
79  {
80  QString field = "field" + QString::number(i +1);
81  if (fields.contains(field))
82  {
83  QString value = fields.value(field).toString().trimmed();
84  if (!value.isEmpty())
85  {
86  m_fields.insert(value, i);
87  m_reverseFields.insert(i, value);
88  }
89  }
90  }
91  }
92 
93  if (m_fields.isEmpty())
94  {
95  LOG(VB_GENERAL, LOG_ERR, QString("%1 logger has no valid fields - disabling").arg(m_description));
96  return false;
97  }
98 
99  // at this point, all should be good. Set up the timer
100  connect(this, SIGNAL(TryNotify()), this, SLOT(DoNotify()));
101  m_timer.setSingleShot(true);
102  connect(this, SIGNAL(StartTimer(int)), &m_timer, SLOT(start(int)));
103  connect(&m_timer, SIGNAL(timeout()), this, SLOT(DoNotify()));
104 
105  return true;
106 }

143 void TorcIOTLogger::DoNotify(void)
144 {
145  if (!GetValid())
146  return;
147 
148  if (!TorcNetwork::IsAvailable())
149  {
150  LOG(VB_GENERAL, LOG_INFO, QString("Not sending %1 update. Network is not available").arg(m_description));
151  return;
152  }
153 
154  QDateTime now = QDateTime::currentDateTime();
155 
156  // ThingSpeak/IOTPlotter etc limit updates to a max of 1 every 15 seconds. If we try and update too fast, simply
157  // amalgamate fields and start the timer to trigger the next update. If the same field is updated more than
158  // once, its value is overwritten with the most recent value. The timestamp may also be out of date for 'stale' fields.
159 
160  if (m_lastUpdate.secsTo(now) < m_maxUpdateInterval)
161  {
162  if (m_timer.isActive())
163  return;
164 
165  StartTimer(m_lastUpdate.msecsTo(now));
166  return;
167  }
168 
169  m_lastUpdate = now;
170 
171  TorcNetworkRequest* request = CreateRequest();
172  if (request)
173  {
174  TorcNetwork::GetAsynchronous(request, this);
175  {
176  QMutexLocker locker(&lock);
177  m_requests.append(request);
178  }
179  }
180 }

182 void TorcIOTLogger::RequestReady(TorcNetworkRequest *Request)
183 {
184  if (!Request)
185  return;
186 
187  QMutexLocker locker(&lock);
188  if (!m_requests.contains(Request))
189  {
190  LOG(VB_GENERAL, LOG_WARNING, QString("Response to unknown %1 request").arg(m_description));
191  return;
192  }
193 
194  int status = Request->GetStatus();
195  if (status >= HTTP_BadRequest)
196  {
197  // not sure whether it is my network or the thingspeak server - but I get intermittent Host Not Found errors
198  // which end up disabling ThingSpeak. So check for HostNotFound...
199  LOG(VB_GENERAL, LOG_ERR, QString("%2 update error '%1'").arg(TorcHTTPRequest::StatusToString((HTTPStatus)status)).arg(m_description));
200 
201  if (Request->GetReplyError() != QNetworkReply::HostNotFoundError)
202  {
203  if (++m_badRequestCount > m_maxBadRequests)
204  {
205  LOG(VB_GENERAL, LOG_ERR, QString("%1 %2 update errors. Disabling logger - Check your config.")
206  .arg(m_badRequestCount)
207  .arg(m_description));
208  SetValid(false);
209  }
210  }
211  }
212  else
213  {
214  // reset error count
215  m_badRequestCount = 0;
216  ProcessRequest(Request);
217  }
218 
219  m_requests.removeAll(Request);
220  Request->DownRef();
221 }
