116 void TorcLanguage::SetLanguageCode(const QString &Language)
117 {
118  QWriteLocker locker(&m_lock);
119 
120  // ignore unnecessary changes
121  QLocale locale(Language);
122  if (m_locale == locale)
123  {
124  LOG(VB_GENERAL, LOG_INFO, "Requested language already set - ignoring");
125  return;
126  }
127 
128  // ignore unknown
129  if (!m_languages.contains(locale))
130  {
131  LOG(VB_GENERAL, LOG_INFO, QString("Requested language (%1) not available - ignoring").arg(Language));
132  return;
133  }
134 
135  // set new details
136  m_locale = locale;
137  languageCode = m_locale.name();
138  languageString = m_locale.nativeLanguageName();
139 
140  LOG(VB_GENERAL, LOG_INFO, QString("Language changed: %1 (%2) (%3)(env - %4)")
141  .arg(QLocale::languageToString(m_locale.language()))
142  .arg(QLocale::countryToString(m_locale.country()))
143  .arg(m_locale.name()).arg(qgetenv("LANG").data()));
144 
145  // load the new translation. This will replace the existing translation.
146  // NB it's not clear from the docs whether this is thread safe.
147  // NB this only supports installing a single translation file. So other 'modules' cannot
148  // currently be loaded.
149 
150  QString filename = QString("torc_%1.qm").arg(m_locale.name());
151  if (!m_translator.load(filename, GetTorcTransDir()))
152  LOG(VB_GENERAL, LOG_ERR, QString("Failed to load translation file '%1' from '%2'").arg(filename).arg(GetTorcTransDir()));
153 
154  // notify change
155  emit LanguageCodeChanged(languageCode);
156  emit LanguageStringChanged(languageString);
157 }

239 void TorcLanguage::InitialiseTranslations(void)
240 {
241  QWriteLocker locker(&m_lock);
242 
243  // clear out old (just in case)
244  m_languages.clear();
245 
246  // retrieve list of installed translation files
247  QDir directory(GetTorcTransDir());
248  QStringList files = directory.entryList(QStringList("torc_*.qm"),
249  QDir::Files | QDir::Readable | QDir::NoSymLinks | QDir::NoDotAndDotDot,
250  QDir::Name);
251 
252  // create a reference list
253  LOG(VB_GENERAL, LOG_INFO, QString("Found %1 translations").arg(files.size()));
254  foreach (QString file, files)
255  {
256  file.chop(3);
257  m_languages.append(QLocale(file.mid(5)));
258  }
259 }
