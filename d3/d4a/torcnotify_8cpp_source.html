201 void TorcNotify::Create(const QVariantMap &Details)
202 {
203  QWriteLocker locker(&m_handlerLock);
204 
205  QVariantMap::const_iterator ii = Details.constBegin();
206  for ( ; ii != Details.constEnd(); ++ii)
207  {
208  if (ii.key() == "notify")
209  {
210  QVariantMap notify = ii.value().toMap();
211  QVariantMap::const_iterator i = notify.constBegin();
212  for ( ; i != notify.constEnd(); ++i)
213  {
214  if (i.key() == "notifiers")
215  {
216  QVariantMap notifiers = i.value().toMap();
217  QVariantMap::iterator it = notifiers.begin();
218  for ( ; it != notifiers.end(); ++it)
219  {
220  QVariantMap details = it.value().toMap();
221  if (!details.contains("name"))
222  {
223  LOG(VB_GENERAL, LOG_ERR, QString("Notifier '%1' has no name").arg(it.key()));
224  continue;
225  }
226 
227  TorcNotifierFactory* factory = TorcNotifierFactory::GetTorcNotifierFactory();
228  for ( ; factory; factory = factory->NextFactory())
229  {
230  TorcNotifier* notifier = factory->Create(it.key(), details);
231  if (notifier)
232  {
233  m_notifiers.append(notifier);
234  LOG(VB_GENERAL, LOG_INFO, QString("New notifier '%1'").arg(notifier->GetUniqueId()));
235  break;
236  }
237  }
238  }
239  }
240  else if (i.key() == "notifications")
241  {
242  QVariantMap notifications = i.value().toMap();
243  QVariantMap::iterator it = notifications.begin();
244  for ( ; it != notifications.end(); ++it)
245  {
246  QVariantMap details = it.value().toMap();
247  if (!details.contains("name"))
248  {
249  LOG(VB_GENERAL, LOG_ERR, QString("Notification '%1' has no name").arg(it.key()));
250  continue;
251  }
252 
253  if (!details.contains("outputs"))
254  {
255  LOG(VB_GENERAL, LOG_ERR, QString("Notification '%1' does not specify 'outputs' (notififiers)").arg(it.key()));
256  continue;
257  }
258 
259  if (!details.contains("message"))
260  {
261  LOG(VB_GENERAL, LOG_ERR, QString("Notificaiton '%1' does not specify 'message'").arg(it.key()));
262  continue;
263  }
264 
265  TorcNotificationFactory* factory = TorcNotificationFactory::GetTorcNotificationFactory();
266  for ( ; factory; factory = factory->NextFactory())
267  {
268  TorcNotification* notification = factory->Create(it.key(), details);
269  if (notification)
