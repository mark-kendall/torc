92 TorcPower* TorcPowerFactory::CreatePower()
93 {
94  TorcPower *power = NULL;
95 
96  int score = 0;
97  TorcPowerFactory* factory = TorcPowerFactory::GetTorcPowerFactory();
98  for ( ; factory; factory = factory->NextPowerFactory())
99  (void)factory->Score(score);
100 
101  factory = TorcPowerFactory::GetTorcPowerFactory();
102  for ( ; factory; factory = factory->NextPowerFactory())
103  {
104  power = factory->Create(score);
105  if (power)
106  break;
107  }
108 
109  if (!power)
110  LOG(VB_GENERAL, LOG_ERR, "Failed to create power implementation");
111 
112  return power;
113 }

304 void TorcPower::BatteryUpdated(int Level)
305 {
306  QMutexLocker locker(&m_lock);
307 
308  if (m_lastBatteryLevel == Level)
309  return;
310 
311  bool wasalreadylow = m_lastBatteryLevel >= 0 && m_lastBatteryLevel <= BatteryLow;
312  m_lastBatteryLevel = Level;
313 
314  if (m_lastBatteryLevel == ACPower)
315  LOG(VB_GENERAL, LOG_INFO, "On AC power");
316  else if (m_lastBatteryLevel == UnknownPower)
317  LOG(VB_GENERAL, LOG_INFO, "Unknown power status");
318  else
319  LOG(VB_GENERAL, LOG_INFO, QString("Battery level %1%").arg(m_lastBatteryLevel));
320 
321  if (!wasalreadylow && (m_lastBatteryLevel >= 0 && m_lastBatteryLevel <= BatteryLow))
322  LowBattery();
323 
324  emit BatteryLevelChanged(m_lastBatteryLevel);
325 }
