294 void TorcCameraVideoOutput::SegmentReady(int Segment)
295 {
296  LOG(VB_GENERAL, LOG_DEBUG, QString("Segment %1 ready").arg(Segment));
297 
298  // allow remote clients to start reading once the first segment is saved
299  m_threadLock.lockForRead();
300  if (!m_cameraStartTime.isValid())
301  {
302  m_threadLock.unlock();
303  m_threadLock.lockForWrite();
304  // set start time now and 'backdate' as the init segment is usually sent once processing has started, so the rest
305  // of the segment arrives quicker than 'expected'
306  m_cameraStartTime = QDateTime::currentDateTimeUtc().addMSecs(VIDEO_SEGMENT_TARGET * -1000);
307  LOG(VB_GENERAL, LOG_INFO, "First segment ready - start time set");
308  }
309  m_threadLock.unlock();
310 
311  QWriteLocker locker(&m_segmentLock);
312  if (!m_segments.contains(Segment))
313  {
314  if (!m_segments.isEmpty() && m_segments.head() >= Segment)
315  {
316  LOG(VB_GENERAL, LOG_ERR, QString("Segment %1 is not greater than head (%2)").arg(Segment).arg(m_segments.head()));
317  }
318  else
319  {
320  m_segments.enqueue(Segment);
321  }
322  }
323  else
324  {
325  LOG(VB_GENERAL, LOG_ERR, QString("Already have a segment #%1").arg(Segment));
326  }
327 }

348 void TorcCameraVideoOutput::ProcessHTTPRequest(const QString &PeerAddress, int PeerPort, const QString &LocalAddress, int LocalPort, TorcHTTPRequest &Request)
349 {
350  (void)PeerAddress;
351  (void)PeerPort;
352  (void)LocalAddress;
353  (void)LocalPort;
354  (void)Request;
355 
356  // treat camera feeds as secure - or maybe not - many players just don't seem to expect authorisation
357  //if (!MethodIsAuthorised(Request, HTTPAuth))
358  // return;
359 
360  // cameraStartTime is set when the first segment has been received
361  m_threadLock.lockForRead();
362  bool valid = m_cameraStartTime.isValid();
363  m_threadLock.unlock();
364  if (!valid)
365  return;
366 
367  Request.SetAllowCORS(true); // needed for a number of browser players.
368  QString method = Request.GetMethod();
369  HTTPRequestType type = Request.GetHTTPRequestType();
370 
371  bool hlsmaster = method.compare(HLS_PLAYLIST_MAST) == 0;
372  bool hlsplaylist = method.compare(HLS_PLAYLIST) == 0;
373  bool player = method.compare(VIDEO_PAGE) == 0;
374  bool dash = method.compare(DASH_PLAYLIST) == 0;
375  bool segment = method.startsWith("segment") && method.endsWith(".m4s");
376  bool init = method.startsWith("init") && method.endsWith(".mp4");
377 
378  if (!(hlsplaylist || player || hlsmaster || dash || segment || init))
379  {
380  Request.SetStatus(HTTP_NotFound);
381  Request.SetResponseType(HTTPResponseDefault);
382  return;
383  }
384 
385  if (type == HTTPOptions)
386  {
387  HandleOptions(Request, HTTPHead | HTTPGet | HTTPOptions);
388  if (segment || init)
389  Request.SetResponseType(HTTPResponseMP4);
390  else if (hlsmaster || hlsplaylist)
391  Request.SetResponseType(HTTPResponseM3U8Apple);
392  else if (player)
393  Request.SetResponseType(HTTPResponseHTML);
394  else
395  Request.SetResponseType(HTTPResponseMPD);
396  return;
397  }
398 
399  if (type != HTTPGet && type != HTTPHead)
400  {
401  Request.SetStatus(HTTP_BadRequest);
402  Request.SetResponseType(HTTPResponseDefault);
403  return;
404  }
405 
406  QByteArray result;
407 
408  if (hlsmaster)
409  {
410  LOG(VB_GENERAL, LOG_DEBUG, "Sending master HLS playlist");
411  result = GetMasterPlaylist();
412  Request.SetAllowGZip(true);
413  Request.SetResponseType(HTTPResponseM3U8Apple);
414  }
415  else if (hlsplaylist)
416  {
417  LOG(VB_GENERAL, LOG_DEBUG, "Sending HLS playlist");
418  result = GetHLSPlaylist();
419  Request.SetAllowGZip(true);
420  Request.SetResponseType(HTTPResponseM3U8Apple);
421  }
422  else if (player)
423  {
424  LOG(VB_GENERAL, LOG_DEBUG, "Sending video page");
425  result = GetPlayerPage();
426  Request.SetAllowGZip(true);
427  Request.SetResponseType(HTTPResponseHTML);
428  }
429  else if (dash)
430  {
431  LOG(VB_GENERAL, LOG_DEBUG, "Sending DASH playlist");
432  result = GetDashPlaylist();
433  Request.SetAllowGZip(true);
434  Request.SetResponseType(HTTPResponseMPD);
435  }
436  else if (segment)
437  {
438  QString number = method.mid(7);
439  number.chop(4);
440  bool ok = false;
441  int num = number.toInt(&ok);
442  if (ok)
443  {
444  QReadLocker locker(&m_threadLock);
445  LOG(VB_GENERAL, LOG_DEBUG, QString("Segment %1 requested").arg(num));
446  result = m_thread->GetSegment(num);
447  if (!result.isEmpty())
448  {
449  Request.SetAllowGZip(false);
450  Request.SetResponseType(HTTPResponseMP4);
451  }
452  else
453  {
454  QReadLocker locker(&m_segmentLock);
455  if (m_segments.isEmpty())
456  {
457  LOG(VB_GENERAL, LOG_WARNING, QString("No segments - %1 requested").arg(num));
458  }
459  else
460  {
461  m_threadLock.lockForRead();
462  QDateTime start = m_cameraStartTime;
463  m_threadLock.unlock();
464  LOG(VB_GENERAL, LOG_WARNING, QString("Segment %1 not found - we have %2-%3")
465  .arg(num).arg(m_segments.first()).arg(m_segments.last()));
466  LOG(VB_GENERAL, LOG_WARNING, QString("Our start time: %1").arg(start.toString(Qt::ISODate)));
467  LOG(VB_GENERAL, LOG_WARNING, QString("Start+request : %1").arg(start.addSecs(num *2).toString(Qt::ISODate)));
468  LOG(VB_GENERAL, LOG_WARNING, QString("Start+first : %1").arg(start.addSecs(m_segments.first() * 2).toString(Qt::ISODate)));
469  LOG(VB_GENERAL, LOG_WARNING, QString("Start+last : %1").arg(start.addSecs(m_segments.last() * 2).toString(Qt::ISODate)));
470  LOG(VB_GENERAL, LOG_WARNING, QString("System time : %1").arg(QDateTime::currentDateTimeUtc().toString(Qt::ISODate)));
471  emit CheckTime();
472  }
473 
474  }
475  }
476  else
477  {
478  Request.SetStatus(HTTP_BadRequest);
479  Request.SetResponseType(HTTPResponseDefault);
480  return;
481  }
482  }
483  else if (init)
484  {
485  QReadLocker locker(&m_threadLock);
486  LOG(VB_GENERAL, LOG_DEBUG, "Init segment requested");
487  result = m_thread->GetInitSegment();
488  if (!result.isEmpty())
489  {
490  Request.SetAllowGZip(false);
491  Request.SetResponseType(HTTPResponseMP4);
492  }
493  else
494  {
495  LOG(VB_GENERAL, LOG_WARNING, "Init segment not found");
496  }
497  }
498 
499  if (!result.isEmpty())
500  {
501  Request.SetResponseContent(result);
502  Request.SetStatus(HTTP_OK);
503  }
504  else
505  {
506  Request.SetStatus(HTTP_NotFound);
507  Request.SetResponseType(HTTPResponseDefault);
508  }
509 }

608 void TorcCameraOutputs::Create(const QVariantMap &Details)
609 {
610  QWriteLocker locker(&m_handlerLock);
611 
612  QVariantMap::const_iterator ii = Details.constBegin();
613  for ( ; ii != Details.constEnd(); ++ii)
614  {
615  if (ii.key() != "outputs")
616  continue;
617 
618  QVariantMap outputs = ii.value().toMap();
619  QVariantMap::const_iterator i = outputs.constBegin();
620  for ( ; i != outputs.constEnd(); ++i)
621  {
622  if (i.key() != "cameras")
623  continue;
624 
625  QVariantMap cameras = i.value().toMap();
626  QVariantMap::iterator it = cameras.begin();
627  for ( ; it != cameras.end(); ++it)
628  {
629  QVariantMap types = it.value().toMap();
630  QVariantMap::iterator it2 = types.begin();
631  for ( ; it2 != types.end(); ++it2)
632  {
633  bool video = false;
634  bool stills = false;
635  QVariantMap camera;
636  if (it2.key() == "video")
637  {
638  video = true;
639  camera = it2.value().toMap();
640  }
641  else if (it2.key() == "stills")
642  {
643  stills = true;
644  camera = it2.value().toMap();
645  }
646  else
647  {
648  LOG(VB_GENERAL, LOG_ERR, QString("Unknown camera interface '%1'").arg(it2.key()));
649  continue;
650  }
651 
652  if (!camera.contains("name"))
653  {
654  LOG(VB_GENERAL, LOG_ERR, QString("Camera '%1' has no name").arg(it.key()));
655  continue;
656  }
657  if (!camera.contains("width"))
658  {
659  LOG(VB_GENERAL, LOG_ERR, QString("Camera '%1' does not specify width").arg(camera.value("name").toString()));
660  continue;
661  }
662  if (!camera.contains("height"))
663  {
664  LOG(VB_GENERAL, LOG_ERR, QString("Camera '%1' does not specify height").arg(camera.value("name").toString()));
665  continue;
666  }
667  bool bitrate = camera.contains("bitrate");
668  bool framerate = camera.contains("framerate");
669 
670  if (video && !bitrate)
671  {
672  LOG(VB_GENERAL, LOG_ERR, QString("Camera video interface '%1' does not specify bitrate").arg(camera.value("name").toString()));
673  continue;
674  }
675 
676  if (video && !framerate)
677  {
678  LOG(VB_GENERAL, LOG_ERR, QString("Camera video interface '%1' does not specify framerate").arg(camera.value("name").toString()));
679  continue;
680  }
681 
682  // NB TorcCameraFactory checks that the underlying class can handle the specified camera
683  // which ensures the TorcCameraOutput instance will be able to create the camera.
684  TorcCameraOutput *newcamera = NULL;
685  TorcCameraFactory* factory = TorcCameraFactory::GetTorcCameraFactory();
686  TorcCameraParams params(Details);
687  for ( ; factory; factory = factory->NextFactory())
688  {
689  if (factory->CanHandle(it.key(), params))
690  {
691  if (video)
692  newcamera = new TorcCameraVideoOutput(it.key(), camera);
693  if (stills)
694  newcamera = new TorcCameraStillsOutput(it.key(), camera);
695 
696  if (newcamera)
697  {
698  m_cameras.insertMulti(it.key(), newcamera);
699  LOG(VB_GENERAL, LOG_INFO, QString("New '%1' camera '%2'").arg(it.key()).arg(newcamera->GetUniqueId()));
700  break;
701  }
702  }
703  }
704  if (NULL == newcamera)
705  LOG(VB_GENERAL, LOG_WARNING, QString("Failed to find handler for camera '%1'").arg(it.key()));
706  }
707  }
708  }
709  }
710 
711  // we now need to analyse our list of cameras and combine the parameters for output devices that
712  // reference the same camera device
713  QList<QString> keys = m_cameras.uniqueKeys();
714  foreach (QString key, keys)
715  {
716  QList<TorcCameraOutput*> outputs = m_cameras.values(key);
717  TorcCameraParams params;
718  foreach (TorcCameraOutput* output, outputs)
719  params = params.Combine(output->GetParams());
720  foreach (TorcCameraOutput* output, outputs)
721  output->SetParams(params);
722  }
723 }

732  it.value()->DownRef();
733  }
