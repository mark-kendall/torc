158 void TorcCameraOutput::ProcessHTTPRequest(const QString &PeerAddress, int PeerPort, const QString &LocalAddress, int LocalPort, TorcHTTPRequest &Request)
159 {
160  (void)PeerAddress;
161  (void)PeerPort;
162  (void)LocalAddress;
163  (void)LocalPort;
164  (void)Request;
165 
166  // treat camera feeds as secure - or maybe not - many players just don't seem to expect authorisation
167  //if (!MethodIsAuthorised(Request, HTTPAuth))
168  // return;
169 
170  // cameraStartTime is set when the camera thread has started (and set to invalid when stopped)
171  if (!m_cameraStartTime.isValid())
172  return;
173 
174  Request.SetAllowCORS(true); // needed for a number of browser players.
175  QString method = Request.GetMethod();
176  HTTPRequestType type = Request.GetHTTPRequestType();
177 
178  bool hlsmaster = method.compare(HLS_PLAYLIST_MAST) == 0;
179  bool hlsplaylist = method.compare(HLS_PLAYLIST) == 0;
180  bool player = method.compare(VIDEO_PAGE) == 0;
181  bool dash = method.compare(DASH_PLAYLIST) == 0;
182  bool segment = method.startsWith("segment") && method.endsWith(".m4s");
183  bool init = method.startsWith("init") && method.endsWith(".mp4");
184 
185  if (!(hlsplaylist || player || hlsmaster || dash || segment || init))
186  {
187  Request.SetStatus(HTTP_NotFound);
188  Request.SetResponseType(HTTPResponseDefault);
189  return;
190  }
191 
192  if (type == HTTPOptions)
193  {
194  HandleOptions(Request, HTTPHead | HTTPGet | HTTPOptions);
195  if (segment || init)
196  Request.SetResponseType(HTTPResponseMP4);
197  else if (hlsmaster || hlsplaylist)
198  Request.SetResponseType(HTTPResponseM3U8Apple);
199  else if (player)
200  Request.SetResponseType(HTTPResponseHTML);
201  else
202  Request.SetResponseType(HTTPResponseMPD);
203  return;
204  }
205 
206  if (type != HTTPGet && type != HTTPHead)
207  {
208  Request.SetStatus(HTTP_BadRequest);
209  Request.SetResponseType(HTTPResponseDefault);
210  return;
211  }
212 
213  QByteArray result;
214 
215  if (hlsmaster)
216  {
217  LOG(VB_GENERAL, LOG_DEBUG, "Sending master HLS playlist");
218  result = GetMasterPlaylist();
219  Request.SetAllowGZip(true);
220  Request.SetResponseType(HTTPResponseM3U8Apple);
221  }
222  else if (hlsplaylist)
223  {
224  LOG(VB_GENERAL, LOG_DEBUG, "Sending HLS playlist");
225  result = GetHLSPlaylist();
226  Request.SetAllowGZip(true);
227  Request.SetResponseType(HTTPResponseM3U8Apple);
228  }
229  else if (player)
230  {
231  LOG(VB_GENERAL, LOG_DEBUG, "Sending video page");
232  result = GetPlayerPage();
233  Request.SetAllowGZip(true);
234  Request.SetResponseType(HTTPResponseHTML);
235  }
236  else if (dash)
237  {
238  LOG(VB_GENERAL, LOG_DEBUG, "Sending DASH playlist");
239  result = GetDashPlaylist();
240  Request.SetAllowGZip(true);
241  Request.SetResponseType(HTTPResponseMPD);
242  }
243  else if (segment)
244  {
245  QString number = method.mid(7);
246  number.chop(4);
247  bool ok = false;
248  int num = number.toInt(&ok);
249  if (ok)
250  {
251  QReadLocker locker(&m_threadLock);
252  LOG(VB_GENERAL, LOG_DEBUG, QString("Segment %1 requested").arg(num));
253  result = m_thread->GetSegment(num);
254  if (!result.isEmpty())
255  {
256  Request.SetAllowGZip(false);
257  Request.SetResponseType(HTTPResponseMP4);
258  }
259  else
260  {
261  LOG(VB_GENERAL, LOG_WARNING, "Segment not found");
262  }
263  }
264  else
265  {
266  Request.SetStatus(HTTP_BadRequest);
267  Request.SetResponseType(HTTPResponseDefault);
268  return;
269  }
270  }
271  else if (init)
272  {
273  QReadLocker locker(&m_threadLock);
274  LOG(VB_GENERAL, LOG_DEBUG, "Init segment requested");
275  result = m_thread->GetInitSegment();
276  if (!result.isEmpty())
277  {
278  Request.SetAllowGZip(false);
279  Request.SetResponseType(HTTPResponseMP4);
280  }
281  else
282  {
283  LOG(VB_GENERAL, LOG_WARNING, "Init segment not found");
284  }
285  }
286 
287  if (!result.isEmpty())
288  {
289  Request.SetResponseContent(result);
290  Request.SetStatus(HTTP_OK);
291  }
292  else
293  {
294  Request.SetStatus(HTTP_NotFound);
295  Request.SetResponseType(HTTPResponseDefault);
296  }
297 }

384 void TorcCameraOutputs::Create(const QVariantMap &Details)
385 {
386  QMutexLocker locker(&m_lock);
387 
388  QVariantMap::const_iterator ii = Details.constBegin();
389  for ( ; ii != Details.constEnd(); ++ii)
390  {
391  if (ii.key() == "outputs")
392  {
393  QVariantMap outputs = ii.value().toMap();
394  QVariantMap::const_iterator i = outputs.constBegin();
395  for ( ; i != outputs.constEnd(); ++i)
396  {
397  if (i.key() == "cameras")
398  {
399  QVariantMap cameras = i.value().toMap();
400  QVariantMap::iterator it = cameras.begin();
401  for ( ; it != cameras.end(); ++it)
402  {
403  QVariantMap camera = it.value().toMap();
404  if (!camera.contains("name"))
405  {
406  LOG(VB_GENERAL, LOG_ERR, QString("Camera '%1' has no name").arg(it.key()));
407  continue;
408  }
409  if (!camera.contains("width"))
410  {
411  LOG(VB_GENERAL, LOG_ERR, QString("Camera '%1' does not specify width").arg(camera.value("name").toString()));
412  continue;
413  }
414  if (!camera.contains("height"))
415  {
416  LOG(VB_GENERAL, LOG_ERR, QString("Camera '%1' does not specify height").arg(camera.value("name").toString()));
417  continue;
418  }
419  if (!camera.contains("bitrate"))
420  {
421  LOG(VB_GENERAL, LOG_ERR, QString("Camera '%1' does not specify bitrate").arg(camera.value("name").toString()));
422  continue;
423  }
424  if (!camera.contains("framerate"))
425  {
426  LOG(VB_GENERAL, LOG_ERR, QString("Camera '%1' does not specify framerate").arg(camera.value("name").toString()));
427  continue;
428  }
429 
430  // NB TorcCameraFactory checks that the underlying class can handle the specified camera
431  // which ensures the TorcCameraOutput instance will be able to create the camera.
432  TorcCameraOutput *newcamera = NULL;
433  TorcCameraFactory* factory = TorcCameraFactory::GetTorcCameraFactory();
434  TorcCameraParams params(Details);
435  for ( ; factory; factory = factory->NextFactory())
436  {
437  if (factory->CanHandle(it.key(), params))
438  {
439  newcamera = new TorcCameraOutput(it.key(), camera);
440  m_cameras.insertMulti(it.key(), newcamera);
441  LOG(VB_GENERAL, LOG_INFO, QString("New '%1' camera '%2'").arg(it.key()).arg(newcamera->GetUniqueId()));
442  break;
443  }
444  }
445  if (NULL == newcamera)
446  LOG(VB_GENERAL, LOG_WARNING, QString("Failed to find handler for camera '%1'").arg(it.key()));
447  }
448  }
449  }
450  }
451  }
452 }
