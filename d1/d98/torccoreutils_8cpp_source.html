63 quint64 TorcCoreUtils::GetMicrosecondCount(void)

67  LARGE_INTEGER ticks;
68  if (QueryPerformanceFrequency(&ticksPerSecond))

132 QByteArray TorcCoreUtils::GZipCompress(QByteArray &Source)
133 {
134  QByteArray result;
135 
136 #ifndef USING_ZLIB
137  (void) Source;
138  return result;
139 #else
140  // this shouldn't happen
141  if (Source.size() < 0)
142  return result;
143 
144  // initialise zlib (see zlib usage examples)
145  static const int chunksize = 16384;
146  char outputbuffer[chunksize];
147 
148  z_stream stream;
149  stream.zalloc = NULL;
150  stream.zfree = NULL;
151  stream.opaque = NULL;
152  stream.avail_in = Source.size();
153  stream.next_in = (Bytef*)Source.data();
154 
155  if (Z_OK != deflateInit2(&stream, Z_DEFAULT_COMPRESSION, Z_DEFLATED, 15 + 16, 8, Z_DEFAULT_STRATEGY))
156  {
157  LOG(VB_GENERAL, LOG_ERR, "Failed to setup zlip decompression");
158  return result;
159  }
160 
161  do
162  {
163  stream.avail_out = chunksize;
164  stream.next_out = (Bytef*)outputbuffer;
165 
166  int error = deflate(&stream, Z_FINISH);
167 
168  if (Z_NEED_DICT == error || error < Z_OK)
169  {
170  LOG(VB_GENERAL, LOG_ERR, "Failed to compress data");
171  deflateEnd(&stream);
172  return result;
173  }
174 
175  result.append(outputbuffer, chunksize - stream.avail_out);
176  } while (stream.avail_out == 0);
177 
178  deflateEnd(&stream);
179  return result;
180 #endif
181 }

188 QByteArray TorcCoreUtils::GZipCompressFile(QFile &Source)
189 {
190  QByteArray result;
191 
192 #ifndef USING_ZLIB
193  (void) Source;
194  return result;
195 #else
196  // this shouldn't happen
197  if (Source.size() < 0)
198  return result;
199 
200  // initialise zlib (see zlib usage examples)
201  static const int chunksize = 32768;
202  char inputbuffer[chunksize];
203  char outputbuffer[chunksize];
204 
205  z_stream stream;
206  stream.zalloc = NULL;
207  stream.zfree = NULL;
208  stream.opaque = NULL;
209 
210  if (Z_OK != deflateInit2(&stream, Z_DEFAULT_COMPRESSION, Z_DEFLATED, 15 + 16, 8, Z_DEFAULT_STRATEGY))
211  {
212  LOG(VB_GENERAL, LOG_ERR, "Failed to setup zlip decompression");
213  return result;
214  }
215 
216  // this should have been checked already
217  if (!Source.isOpen())
218  if (!Source.open(QIODevice::ReadOnly))
219  return result;
220 
221  while (Source.bytesAvailable() > 0)
222  {
223  stream.avail_out = chunksize;
224  stream.next_out = (Bytef*)outputbuffer;
225 
226  qint64 read = Source.read(inputbuffer, qMin(Source.bytesAvailable(), (qint64)chunksize));
227 
228  if (read > 0)
229  {
230  stream.avail_in = read;
231  stream.next_in = (Bytef*)inputbuffer;
232 
233  int error = deflate(&stream, Source.bytesAvailable() ? Z_SYNC_FLUSH : Z_FINISH);
234 
235  if (error == Z_OK || error == Z_STREAM_END)
236  {
237  result.append(outputbuffer, chunksize - stream.avail_out);
238  continue;
239  }
240  }
241 
242  LOG(VB_GENERAL, LOG_ERR, "Failed to compress file");
243  deflateEnd(&stream);
244  return result;
245  }
246 
247  deflateEnd(&stream);
248  return result;
249 #endif
250 }
