43 TorcPiPWMOutput::TorcPiPWMOutput(int Pin, const QVariantMap &Details)
44  : TorcPWMOutput(DEFAULT_VALUE, "PiGPIOPWMOutput", Details, PI_PWM_RESOLUTION),
45  m_pin(Pin)
46 {
47  // hardware PWM only operates at default 10bit accuracy
48  if ((m_resolution != m_maxResolution) && (m_pin == TORC_HWPWM_PIN))
49  {
50  m_resolution = m_maxResolution;
51  LOG(VB_GENERAL, LOG_WARNING, QString("Ignoring user defined resolution for hardware PWM - defaulting to %1").arg(m_maxResolution));
52  }
53 
54  int value = (int)((DEFAULT_VALUE * (double)m_resolution) + 0.5);
55 
56  if (m_pin == TORC_HWPWM_PIN)
57  {
58  pinMode(m_pin, PWM_OUTPUT);
59  pwmWrite(m_pin, value);
60  }
61  else
62  {
63  if (softPwmCreate(m_pin, value, m_resolution) == 0)
64  {
65  LOG(VB_GENERAL, LOG_WARNING, QString("Using software PWM on pin %1: It MIGHT flicker...").arg(m_pin));
66  }
67  else
68  {
69  LOG(VB_GENERAL, LOG_ERR, QString("Failed to setup software PWM on pin %1").arg(m_pin));
70  }
71  }
72 }
