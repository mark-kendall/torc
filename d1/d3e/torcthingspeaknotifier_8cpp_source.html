43 void TorcThingSpeakNotifier::ProcessRequest(TorcNetworkRequest *Request)
44 {
45  if (!Request)
46  return;
47 
48  // a successful update returns a non-zero update id - and zero on error.
49  QByteArray result = Request->ReadAll(1000).trimmed();
50  bool ok = false;
51  quint64 id = result.toLongLong(&ok);
52  if (ok)
53  {
54  if (id == 0)
55  LOG(VB_GENERAL, LOG_ERR, QString("%1 update failed for '%2'").arg(m_description, uniqueId));
56  }
57  else
58  {
59  LOG(VB_GENERAL, LOG_ERR, QString("Failed to parse update id from %1 response (%2)").arg(m_description, uniqueId));
60  }
61 }

63 TorcNetworkRequest* TorcThingSpeakNotifier::CreateRequest(void)
64 {
65  QUrl url(THINGSPEAK_UPDATE_URL);
66  QUrlQuery query;
67  query.addQueryItem("api_key", m_apiKey);
68  for (int i = 0; i < m_maxFields; i++)
69  {
70  if (!m_fieldValues[i].isEmpty())
71  query.addQueryItem(QString("field%1").arg(i+1), m_fieldValues[i]);
72  // reset the fields
73  m_fieldValues[i] = QString("");
74  }
75  url.setQuery(query);
76  QNetworkRequest qrequest(url);
77 
78  return new TorcNetworkRequest(qrequest, QNetworkAccessManager::GetOperation, 0, &m_networkAbort);
79 }
