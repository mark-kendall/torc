73 TorcCentral::TorcCentral()
74  : QObject(),
75  TorcHTTPService(this, "central", "central", TorcCentral::staticMetaObject, ""),
76  m_config(QVariantMap()),
77  m_graph(),
78  temperatureUnits("celsius")
79 {
80  // reset state graph and clear out old files
81  // content directory should already have been created by TorcHTMLDynamicContent
82  QString graphdot = GetTorcContentDir() + "stategraph.dot";
83  QString graphsvg = GetTorcContentDir() + "stategraph.svg";
84  QString config = GetTorcContentDir() + TORC_CONFIG_FILE;
85  QString current = GetTorcConfigDir() + "/" + TORC_CONFIG_FILE;
86 
87  if (QFile::exists(graphdot))
88  QFile::remove(graphdot);
89  if (QFile::exists(graphsvg))
90  QFile::remove(graphsvg);
91  if (QFile::exists(config))
92  QFile::remove(config);
93 
94  QFile currentconfig(current);
95  if (!currentconfig.copy(config))
96  LOG(VB_GENERAL, LOG_WARNING, "Failed to copy current config file to content directory");
97 
98  // listen for interesting events
99  gLocalContext->AddObserver(this);
100 
101  if (LoadConfig())
102  {
103  TemperatureUnits temperatureunits = Celsius; // default to metric
104 
105  // handle settings now
106  if (m_config.contains("settings"))
107  {
108  QVariantMap settings = m_config.value("settings").toMap();
109 
110  // applicationname
111  if (settings.contains("applicationname"))
112  {
113  QString name = settings.value("applicationname").toString().trimmed();
114  if (!name.isEmpty())
115  {
116  QCoreApplication::setApplicationName(name);
117  LOG(VB_GENERAL, LOG_INFO, QString("Changed application name to '%1'").arg(name));
118  }
119  }
120 
121  // temperature units - metric or imperial...
122  if (settings.contains("temperatureunits"))
123  {
124  QString units = settings.value("temperatureunits").toString().trimmed().toLower();
125  if (units == "metric" || units == "celsius")
126  temperatureunits = Celsius;
127  else if (units == "imperial" || units == "fahrenheit")
128  temperatureunits = Fahrenheit;
129  else
130  LOG(VB_GENERAL, LOG_WARNING, "Unknown temperature units - defaulting to metric (Celsius)");
131  }
132  }
133 
134  LOG(VB_GENERAL, LOG_INFO, QString("Using '%1' temperature units").arg(TemperatureUnitsToString(temperatureunits)));
135  TorcCentral::gTemperatureUnits = temperatureunits;
136  temperatureUnits = TemperatureUnitsToString(temperatureunits);
137 
138  // create the devices
139  TorcDeviceHandler::Start(m_config);
140 
141  // connect controls to sensors/outputs/other controls
142  TorcControls::gControls->Validate();
143 
144  // setup notifications/notifiers
145  (void)TorcNotify::gNotify->Validate();
146 
147  LOG(VB_GENERAL, LOG_INFO, "Initialising state machine");
148  // initialise the state machine
149  {
150  QMutexLocker lock(TorcDevice::gDeviceListLock);
151 
152  QHash<QString,TorcDevice*>::const_iterator it = TorcDevice::gDeviceList->constBegin();
153  for( ; it != TorcDevice::gDeviceList->constEnd(); ++it)
154  it.value()->Start();
155  }
156 
157  LOG(VB_GENERAL, LOG_INFO, "Notifying start");
158  TorcLocalContext::NotifyEvent(Torc::Start);
159 
160  // iff we have got this far, then create the graph
161  // start the graph
162  m_graph.clear();
163  m_graph.append(QString("strict digraph \"%1\" {\r\n"
164  " rankdir=\"LR\";\r\n"
165  " node [shape=rect];\r\n")
166  .arg(TORC_TORC));
167 
168  // build the graph contents
169  TorcInputs::gInputs->Graph(&m_graph);
170  TorcOutputs::gOutputs->Graph(&m_graph);
171  TorcControls::gControls->Graph(&m_graph);
172  TorcNotify::gNotify->Graph(&m_graph);
173 
174  // complete the graph
175  m_graph.append("}\r\n");
176 
177  QFile file(graphdot);
178  if (file.open(QIODevice::ReadWrite))
179  {
180  file.write(m_graph);
181  file.flush();
182  file.close();
183  LOG(VB_GENERAL, LOG_INFO, QString("Saved state graph as %1").arg(graphdot));
184  }
185  else
186  {
187  LOG(VB_GENERAL, LOG_ERR, QString("Failed to open '%1' to write state graph").arg(graphdot));
188  }
189 
190  // create a representation of the state graph
191 #ifdef USING_GRAPHVIZ_LIBS
192  bool created = false;
193 
194  FILE *handle = fopen(graphsvg.toLocal8Bit().data(), "w");
195  if (handle)
196  {
197  GVC_t *gvc = gvContext();
198  Agraph_t *g = agmemread(m_graph.data());
199  gvLayout(gvc, g, "dot");
200  gvRender(gvc, g, "svg", handle);
201  gvFreeLayout(gvc,g);
202  agclose(g);
203  gvFreeContext(gvc);
204  fclose(handle);
205  created = true;
206  }
207  else
208  {
209  LOG(VB_GENERAL, LOG_WARNING, QString("Failed to open '%1' for writing (err: %2)").arg(graphsvg).arg(strerror(errno)));
210  }
211 
212  if (!created)
213  {
214 #endif
215  // NB QProcess appears to be fatally broken. Just use system instead
216  QString command = QString("dot -Tsvg -o %1 %2").arg(graphsvg).arg(graphdot);
217  int err = system(command.toLocal8Bit());
218  if (err < 0)
219  {
220  LOG(VB_GENERAL, LOG_ERR, QString("Failed to create stategraph representation (err: %1)").arg(strerror(errno)));
221  }
222  else
223  {
224  LOG(VB_GENERAL, LOG_INFO, QString("Saved state graph representation as %1").arg(graphsvg));
225  if (err > 0)
226  LOG(VB_GENERAL, LOG_ERR, "dot returned an unexpected result - stategraph may be incomplete or absent");
227  }
228  }
229 #ifdef USING_GRAPHVIZ_LIBS
230  }
231 #endif
232  // no need for the graph data from here
233  m_graph.clear();
234 }

260 bool TorcCentral::LoadConfig(void)
261 {
262  bool skipvalidation = false;
263 
264  if (!qgetenv("TORC_NO_VALIDATION").isEmpty())
265  {
266  LOG(VB_GENERAL, LOG_INFO, "Skipping configuration file validation (command line).");
267  skipvalidation = true;
268  }
269 
270  QString xml = GetTorcConfigDir() + "/" + TORC_CONFIG_FILE;
271  QFileInfo config(xml);
272  if (!skipvalidation && !config.exists())
273  {
274  LOG(VB_GENERAL, LOG_ERR, QString("Failed to find configuration file '%1'").arg(xml));
275  return false;
276  }
277 
278 #if defined(USING_XMLPATTERNS) || defined(USING_LIBXML2)
279  QString customxsd = GetTorcContentDir() + "torc.xsd";
280  // we always want to delete the old xsd - if it isn't present, it wasn't used!
281  // so retrieve now and then delete
282  QByteArray oldxsd;
283  if (QFile::exists(customxsd))
284  {
285  QFile customxsdfile(customxsd);
286  if (customxsdfile.open(QIODevice::ReadOnly))
287  {
288  oldxsd = customxsdfile.readAll();
289  customxsdfile.close();
290  }
291  QFile::remove(customxsd);
292  }
293 
294  QString basexsd = GetTorcShareDir() + "/html/torc.xsd";
295  if (!QFile::exists(basexsd))
296  {
297  LOG(VB_GENERAL, LOG_ERR, QString("Failed to find base XSD file '%1'").arg(basexsd));
298  return false;
299  }
300 
301  QByteArray newxsd;
302  if (!skipvalidation)
303  {
304  // customise and save the xsd now - even if validation is skipped from here, we need the
305  // xsd for reference
306  newxsd = GetCustomisedXSD(basexsd);
307  if (!newxsd.isEmpty())
308  {
309  // save the XSD for the user to inspect if necessary
310  QFile customxsdfile(customxsd);
311  if (customxsdfile.open(QIODevice::ReadWrite))
312  {
313  customxsdfile.write(newxsd);
314  customxsdfile.flush();
315  customxsdfile.close();
316  LOG(VB_GENERAL, LOG_INFO, QString("Saved current XSD as '%1'").arg(customxsd));
317  }
318  else
319  {
320  LOG(VB_GENERAL, LOG_WARNING, QString("Failed to open '%1' for writing").arg(customxsd));
321  }
322  }
323  else
324  {
325  LOG(VB_GENERAL, LOG_ERR, "Strange - empty xsd...");
326  return false;
327  }
328  }
329 
330  // validation can take a while on slower machines (e.g. single core raspberry pi).
331  // try and skip if the config has not been modified
332  QString lastvalidated = gLocalContext->GetSetting("configLastValidated", QString("never"));
333  if (!skipvalidation && lastvalidated != "never")
334  {
335  bool xsdmodified = qstrcmp(oldxsd.constData(), newxsd.constData()) != 0;
336  bool configmodified = config.lastModified() >= QDateTime::fromString(lastvalidated);
337 
338  if (xsdmodified)
339  LOG(VB_GENERAL, LOG_INFO, QString("XSD file changed since last validation"));
340  else
341  LOG(VB_GENERAL, LOG_INFO, QString("XSD unchanged since last validation"));
342 
343  if (configmodified)
344  LOG(VB_GENERAL, LOG_INFO, QString("Configuration file changed since last validation"));
345  else
346  LOG(VB_GENERAL, LOG_INFO, QString("Configuration file unchanged since last validation:"));
347 
348  skipvalidation = !xsdmodified && !configmodified;
349  }
350 
351  if (!skipvalidation)
352  {
353  LOG(VB_GENERAL, LOG_INFO, "Starting validation of configuration file");
354  TorcXmlValidator validator(xml, newxsd);
355  if (!validator.Validated())
356  {
357  LOG(VB_GENERAL, LOG_ERR, QString("Configuration file '%1' failed validation").arg(xml));
358  // make sure we re-validate
359  gLocalContext->SetSetting("configLastValidated", QString("never"));
360  return false;
361  }
362 
363  LOG(VB_GENERAL, LOG_INFO, "Configuration successfully validated");
364  gLocalContext->SetSetting("configLastValidated", QDateTime::currentDateTime().toString());
365  }
366  else
367  {
368  LOG(VB_GENERAL, LOG_INFO, "Skipping validation of configuration file");
369  }
370 #else
371  LOG(VB_GENERAL, LOG_INFO, "Xml validation unavailable - not validating configuration file.");
372 #endif
373 
374  TorcXMLReader reader(xml);
375  QString error;
376 
377  if (!reader.IsValid(error))
378  {
379  LOG(VB_GENERAL, LOG_ERR, error);
380  return false;
381  }
382 
383  QVariantMap result = reader.GetResult();
384 
385  // root object should be 'torc'
386  if (!result.contains("torc"))
387  {
388  LOG(VB_GENERAL, LOG_ERR, QString("Failed to find 'torc' root element in '%1'").arg(xml));
389  return false;
390  }
391 
392  m_config = result.value("torc").toMap();
393 
394  LOG(VB_GENERAL, LOG_INFO, QString("Loaded config from %1").arg(xml));
395  return true;
396 }

421 bool TorcCentral::event(QEvent *Event)
422 {
423  TorcEvent* torcevent = dynamic_cast<TorcEvent*>(Event);
424  if (torcevent)
425  {
426  int event = torcevent->GetEvent();
427  switch (event)
428  {
429  case Torc::ShuttingDown:
430  case Torc::Suspending:
431  case Torc::Restarting:
432  case Torc::Hibernating:
433  // NB this just ensures outputs (e.g. PWM drivers) are set to sensible
434  // defaults if we shut down. There is currently no handling of waking, which
435  // would need more thought about resetting state etc.
436  LOG(VB_GENERAL, LOG_INFO, "Resetting devices to defaults.");
437  {
438  QMutexLocker lock(TorcDevice::gDeviceListLock);
439 
440  QHash<QString,TorcDevice*>::const_iterator it = TorcDevice::gDeviceList->constBegin();
441  for( ; it != TorcDevice::gDeviceList->constEnd(); ++it)
442  it.value()->Stop();
443  }
444 
445  break;
446  default: break;
447  }
448  }
449 
450  return QObject::event(Event);
451 }
