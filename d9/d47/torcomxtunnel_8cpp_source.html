91 OMX_ERRORTYPE TorcOMXTunnel::Destroy(void)

102 
103  m_source->EnablePort(OMX_DirOutput, m_sourceIndex, false, m_sourceDomain);
104  m_destination->EnablePort(OMX_DirInput, m_destinationIndex, false, m_destinationDomain);
105 
106  OMX_ERRORTYPE error = OMX_SetupTunnel(m_source->GetHandle(), m_sourcePort, NULL, 0);
107  if (OMX_ErrorNone != error)
108  OMX_ERROR(error, m_source->GetName(), "Failed to destroy tunnel input");
109 
110  error = OMX_SetupTunnel(m_destination->GetHandle(), m_destinationPort, NULL, 0);
111  if (OMX_ErrorNone != error)
112  OMX_ERROR(error, m_destination->GetName(), "Failed to destroy tunnel output");
113 
114  m_connected = false;
115  return OMX_ErrorNone;
