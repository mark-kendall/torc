72 void TorcI2CPCA9685Channel::SetValue(double Value)
73 {
74  QMutexLocker locker(&lock);
75 
76  // anything that doesn't change the range sent to the device is just noise, so
77  // calculate now and filter early
78  double newvalue = Value;
79  if (!ValueIsDifferent(newvalue))
80  return;
81 
82  // convert 0.0 to 1.0 to 0 to 4095
83  int channelvalue = (int)((newvalue * (float)PCA9685_RESOLUTION) + 0.5);
84  if (channelvalue == m_channelValue)
85  return;
86 
87  m_channelValue = channelvalue;
88  m_parent->SetPWM(m_channelNumber, m_channelValue);
89  TorcPWMOutput::SetValue(Value);
90 }

92 TorcI2CPCA9685::TorcI2CPCA9685(int Address, const QVariantMap &Details)
93  : TorcI2CDevice(Address)
94 {
95  // nullify outputs in case they aren't created
96  memset(m_outputs, 0, sizeof(TorcI2CPCA9685Channel*) * 16);
97 
98  // open a handle to the device
99  m_fd = wiringPiI2CSetup(m_address);
100  if (m_fd < 0)
101  {
102  LOG(VB_GENERAL, LOG_ERR, QString("Failed to open I2C device at address 0x%1").arg(m_address, 0, 16));
103  return;
104  }
105 
106  LOG(VB_GENERAL, LOG_INFO, QString("Opened %1 I2C device at address 0x%2")
107  .arg(PCA9685).arg(m_address, 0, 16));
108 
109  // reset
110  if (wiringPiI2CWriteReg8(m_fd, MODE1, 0x00) < 0 || wiringPiI2CWriteReg8(m_fd, MODE2, 0x04) < 0)
111  {
112  LOG(VB_GENERAL, LOG_ERR, "Failed to reset PCA9685 device");
113  return;
114  }
115 
116  // set frequency to 1000Hz
117  bool success = true;
118  success &= wiringPiI2CWriteReg8(m_fd, MODE1, 0x01) > -1;
119  success &= wiringPiI2CWriteReg8(m_fd, PRESCALE, (uint8_t)((CLOCKFREQ / PCA9685_RANGE / 1000) - 1)) > -1;
120  success &= wiringPiI2CWriteReg8(m_fd, MODE1, 0x80) > -1;
121  success &= wiringPiI2CWriteReg8(m_fd, MODE2, 0x04) > -1;
122 
123  if (!success)
124  LOG(VB_GENERAL, LOG_ERR, "Failed to set up PCA9685 device");
125 
126  // create individual channel services
127  // this will also reset each channel to the default value (0)
128  QVariantMap::const_iterator it = Details.begin();
129  for ( ; it != Details.constEnd(); ++it)
130  {
131  if (it.key() == "channel")
132  {
133  // channel needs a <number>
134  QVariantMap channel = it.value().toMap();
135  if (!channel.contains("number"))
136  {
137  LOG(VB_GENERAL, LOG_ERR, QString("PCA9685 channel has no number"));
138  continue;
139  }
140 
141  bool ok = false;
142  int channelnum = channel.value("number").toInt(&ok);
143 
144  if (!ok || channelnum < 0 || channelnum >= 16)
145  {
146  LOG(VB_GENERAL, LOG_ERR, QString("Failed to parse valid PCA9685 channel number from '%1'").arg(channel.value("number").toString()));
147  continue;
148  }
149 
150  if (m_outputs[channelnum])
151  {
152  LOG(VB_GENERAL, LOG_ERR, QString("PCA9685 channel '%1' is already in use").arg(channelnum));
153  continue;
154  }
155 
156  m_outputs[channelnum] = new TorcI2CPCA9685Channel(channelnum, this, channel);
157  }
158  }
159 }

181 bool TorcI2CPCA9685::SetPWM(int Channel, int Value)
182 {
183  if (m_fd < 0 || Channel < 0 || Channel > 15)
184  return false;
185 
186  int offtime = qBound(0, Value, PCA9685_RESOLUTION);
187  int ontime = 0;
188 
189  // turn completely on or off if required
190  // 'off' supercedes 'on' per spec
191  if (offtime < 1)
192  {
193  LOG(VB_GENERAL, LOG_INFO, QString("Channel %1 turned completely OFF").arg(Channel));
194  offtime |= 0x1000;
195  }
196  else if (offtime >= PCA9685_RESOLUTION)
197  {
198  LOG(VB_GENERAL, LOG_INFO, QString("Channel %1 turned completely ON").arg(Channel));
199  ontime |= 0x1000;
200  }
201 
202  bool success = true;
203  success &= wiringPiI2CWriteReg8(m_fd, LED0_ON_LOW + (4 * Channel), ontime & 0xFF) > -1;
204  success &= wiringPiI2CWriteReg8(m_fd, LED0_ON_HIGH + (4 * Channel), ontime >> 8) > -1;
205  success &= wiringPiI2CWriteReg8(m_fd, LED0_OFF_LOW + (4 * Channel), offtime & 0xFF) > -1;
206  success &= wiringPiI2CWriteReg8(m_fd, LED0_OFF_HIGH + (4 * Channel), offtime >> 8) > -1;
207 
208  if (!success)
209  LOG(VB_GENERAL, LOG_ERR, QString("Error writing to channel %1").arg(Channel));
210  return success;
211 }
