164 void TorcTriggerNotification::InputValueChanged(double Value)
165 {
166  // N.B. this should be thread safe as InputValueChanged is always triggered via a signal
167  TorcDevice* input = qobject_cast<TorcDevice*>(sender());
168 
169  if (input && (input == m_input))
170  {
171  if ((Value > 0.0 && m_lastValue <= 0.0 && m_triggerHigh) || // low to high
172  (Value <= 0.0 && m_lastValue > 0.0 && !m_triggerHigh)) // high to low
173  {
174  // add 'key'-'value' data for each reference (e.g. 'tanktemperature', '27.0')
175  foreach (TorcDevice* device, m_referenceDevices)
176  m_customData.insert(device->GetUniqueId(), QString::number(device->GetValue()));
177  QVariantMap message = TorcNotify::gNotify->SetNotificationText(m_title, m_body, m_customData);
178  emit Notify(message);
179  }
180 
181  m_lastValue = Value;
182  }
183  else
184  {
185  LOG(VB_GENERAL, LOG_ERR, QString("TriggerNotifiction '%1' signalled from unknown input").arg(uniqueId));
186  }
187 }

193 bool TorcTriggerNotification::Setup(void)
194 {
195  if (m_inputName.isEmpty())
196  return false;
197 
198  // setup the input
199  TorcDevice* input = NULL;
200  {
201  QMutexLocker locker(gDeviceListLock);
202  if (!m_inputName.isEmpty() && gDeviceList->contains(m_inputName))
203  input = gDeviceList->value(m_inputName);
204  }
205 
206  if (!input)
207  {
208  LOG(VB_GENERAL, LOG_ERR, QString("Failed to find input '%1' for trigger notification '%2'").arg(m_inputName).arg(uniqueId));
209  return false;
210  }
211 
212  connect(input, SIGNAL(ValueChanged(double)), this, SLOT(InputValueChanged(double)));
213  m_input = input;
214 
215  // check for the existence of reference devices
216  foreach (QString reference, m_references)
217  {
218  QMutexLocker locker(gDeviceListLock);
219  if (!reference.isEmpty() && gDeviceList->contains(reference))
220  m_referenceDevices.append(gDeviceList->value(reference));
221  }
222 
223  return TorcNotification::Setup();
224 }
