50 void InitialiseTorcDirectories(TorcCommandLine* CommandLine)
51 {
52  static bool initialised = false;
53  if (initialised)
54  return;
55 
56  initialised = true;
57 
58  gInstallDir = QString(PREFIX) + "/";
59  gShareDir = QString(RUNPREFIX) + "/share/" + TORC_TORC;
60  // override shared directory
61  QString sharedir = CommandLine ? CommandLine->GetValue("share").toString() : QString();
62  if (!sharedir.isEmpty())
63  gShareDir = sharedir;
64 
65  gTransDir = gShareDir + "/i18n/";
66  // override translation directory - which may be done independantly of shared directory
67  QString transdir = CommandLine ? CommandLine->GetValue("trans").toString() : QString();
68  if (!transdir.isEmpty())
69  gTransDir = transdir;
70 
71  gConfDir = QDir::homePath() + "/." + TORC_TORC;
72  // override config directory - and by implication the content directory
73  QString confdir = CommandLine ? CommandLine->GetValue("config").toString() : QString();
74  if (!confdir.isEmpty())
75  gConfDir = confdir;
76  gContentDir = gConfDir + TORC_CONTENT_DIR;
77 }
