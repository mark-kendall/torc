51 static void ExitHandler(int Sig)
52 {
53  if (SIGPIPE == Sig)
54  {
55  LOG(VB_GENERAL, LOG_WARNING, "Received SIGPIPE interrupt - ignoring");
56  return;
57  }
58 
59  LOG(VB_GENERAL, LOG_INFO, QString("Received %1")

193 bool TorcLocalContext::Init(void)
194 {
195  // Create the configuration directory
196  QString configdir = GetTorcConfigDir();
197 
198  QDir dir(configdir);
199  if (!dir.exists())
200  {
201  if (!dir.mkpath(configdir))
202  {
203  LOG(VB_GENERAL, LOG_ERR, QString("Failed to create config directory ('%1')")
204  .arg(configdir));
205  return false;
206  }
207  }
208 
209  // Open the local database
210  if (m_dbName.isEmpty())
211  m_dbName = configdir + "/" + TORC_TORC + "-settings.sqlite";
212 
213  m_sqliteDB = new TorcSQLiteDB(m_dbName);
214 
215  if (!m_sqliteDB || !m_sqliteDB->IsValid())
216  return false;
217 
218  // Load the settings
219  {
220  QWriteLocker locker(&m_localSettingsLock);
221  m_sqliteDB->LoadSettings(m_localSettings);
222  }
223 
224  // Create the root settings object
225  gRootSetting = new TorcSettingGroup(nullptr, TORC_ROOT_SETTING);
226 
227  // create/load the UUID - make this persistent to ensure peers do not think
228  // there are multiple devices after a number of restarts.
229  QString uuid = QUuid::createUuid().toString();
230  if (uuid.startsWith('{'))
231  uuid = uuid.mid(1);
232  if (uuid.endsWith('}'))
233  uuid.chop(1);
234  TorcSetting* uuidsaved = new TorcSetting(nullptr, QString("uuid"),QString("UUID"), TorcSetting::String, TorcSetting::Persistent, QVariant(uuid));
235  m_uuid = uuidsaved->GetValue().toString();
236  uuidsaved->Remove();
237  uuidsaved->DownRef();
238  uuidsaved = nullptr;
239 
240  LOG(VB_GENERAL, LOG_INFO, QString("UUID: %1").arg(m_uuid));
241 
242  // Load language and translation preferences
243  m_language = new TorcLanguage(gRootSetting);
244 
245  /* We no longer use QRunnables, so ignore this for now at least
246  // don't expire threads
247  QThreadPool::globalInstance()->setExpiryTimeout(-1);
248 
249  // Increase the thread count
250  int ideal = QThreadPool::globalInstance()->maxThreadCount();
251  int want = ideal * 8;
252  LOG(VB_GENERAL, LOG_INFO, QString("Setting thread pool size to %1 (was %2)").arg(want).arg(ideal));
253  QThreadPool::globalInstance()->setMaxThreadCount(want);
254  */
255 
256  // Bump the UI thread priority
257  QThread::currentThread()->setPriority(QThread::TimeCriticalPriority);
258 
259  // Qt version?
260  LOG(VB_GENERAL, LOG_INFO, QString("Qt runtime version '%1' (compiled with '%2')")
261  .arg(qVersion(), QT_VERSION_STR));
262 
263  // we don't use an admin thread as purely a server (i.e. no gui) - may need to revisit
264 #ifdef TORC_TEST
265  m_adminThread = new TorcAdminThread();
266  m_adminThread->start();
267 #else
268  TorcAdminObject::CreateObjects();
269 #endif
270 
271  return true;
272 }

366 bool TorcLocalContext::QueueShutdownEvent(int Event)
367 {
368  QWriteLocker locker(&m_localSettingsLock);
369 
370  if (m_shutdownDelay < 1)
371  return false;
372 
373  int newevent = Torc::None;
374  switch (Event)
375  {
376  case Torc::RestartTorc:
377  case Torc::Stop:
378  case Torc::Shutdown:
379  case Torc::Restart:
380  case Torc::Hibernate:
381  case Torc::Suspend:
382  newevent = Event;
383  break;
384  default: break;
385  }
386 
387  if (newevent == Torc::None)
388  return false;
389 
390  if (m_shutdownEvent != Torc::None)
391  {
392  LOG(VB_GENERAL, LOG_INFO, QString("Shutdown already queued - ignoring '%1' event").arg(TorcCoreUtils::EnumToString<Torc::Actions>((Torc::Actions)newevent)));
393  return true;
394  }
395 
396  m_shutdownEvent = newevent;
397  LOG(VB_GENERAL, LOG_INFO, QString("Queued '%1' event for %2 seconds").arg(TorcCoreUtils::EnumToString<Torc::Actions>((Torc::Actions)m_shutdownEvent)).arg(m_shutdownDelay));
398  TorcEvent torcevent(Torc::TorcWillStop);
399  Notify(torcevent);
400  QTimer::singleShot(m_shutdownDelay * 1000, this, SLOT(ShutdownTimeout()));
401  return true;
402 }

404 bool TorcLocalContext::HandleShutdown(int Event)
405 {
406  int event = Event == Torc::None ? m_shutdownEvent : Event;
407  switch (event)
408  {
409  case Torc::RestartTorc:
410  LOG(VB_GENERAL, LOG_INFO, "Restarting application");
411  TorcReferenceCounter::EventLoopEnding(true);
412  QCoreApplication::exit(TORC_EXIT_RESTART);
413  return true;
414  case Torc::Stop:
415  LOG(VB_GENERAL, LOG_INFO, "Stopping application");
416  TorcReferenceCounter::EventLoopEnding(true);
417  QCoreApplication::quit();
418  return true;
419  case Torc::Suspend:
420  {
421  TorcEvent event(Torc::SuspendNow);
422  Notify(event);
423  }
424  return true;
425  case Torc::Hibernate:
426  {
427  TorcEvent event(Torc::HibernateNow);
428  Notify(event);
429  }
430  return true;
431  case Torc::Shutdown:
432  {
433  TorcEvent event(Torc::ShutdownNow);
434  Notify(event);
435  }
436  return true;
437  case Torc::Restart:
438  {
439  TorcEvent event(Torc::RestartNow);
440  Notify(event);
441  }
442  return true;
443  default:
444  break;
445  }
446  return false;
447 }
