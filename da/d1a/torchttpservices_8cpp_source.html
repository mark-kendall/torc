79 QVariantMap TorcHTTPServices::ProcessRequest(const QString &Method, const QVariant &Parameters, QObject *Connection, bool Authenticated)
80 {
81  // N.B. this returns a result associated with the actual state of the underlying websocket and
82  // not what the browser etc think is the correct state.
83  if (Method.endsWith("IsSecure"))
84  {
85  QVariant sec((bool)false);
86  TorcWebSocket *connection = qobject_cast<TorcWebSocket*>(Connection);
87  if (connection && connection->IsSecure())
88  sec = true;
89 
90  QVariantMap result;
91  QVariantMap secure;
92  secure.insert("secure", sec);
93  result.insert("result", secure);
94  return result;
95  }
96 
97  return TorcHTTPService::ProcessRequest(Method, Parameters, Connection, Authenticated);
98 }

101 void TorcHTTPServices::ProcessHTTPRequest(const QString &PeerAddress, int PeerPort, const QString &LocalAddress, int LocalPort, TorcHTTPRequest &Request)
102 {
103  // handle own service
104  if (!Request.GetMethod().isEmpty())
105  {
106  // we handle GetWebSocketToken/IsSecure manually as they need access to the authentication headers.
107  // and they have no value internally (hence dummy implementations)
108  QString method = Request.GetMethod();
109  if (method == "GetWebSocketToken" || method == "IsSecure")
110  {
111  HTTPRequestType type = Request.GetHTTPRequestType();
112  if (type == HTTPOptions)
113  {
114  Request.SetStatus(HTTP_OK);
115  Request.SetResponseType(HTTPResponseDefault);
116  Request.SetAllowed(HTTPGet | HTTPOptions);
117  }
118  else if (type == HTTPGet)
119  {
120  QVariant result;
121  QString type;
122  if (method == "GetWebSocketToken")
123  {
124  // force authentication
125  if (!TorcHTTPHandler::MethodIsAuthorised(Request, HTTPAuth))
126  return;
127  result = TorcWebSocketToken::GetWebSocketToken(PeerAddress);
128  type = "accesstoken";
129  }
130  else if (method == "IsSecure")
131  {
132  result = Request.GetSecure();
133  type = "secure";
134  }
135 
136  Request.SetStatus(HTTP_OK);
137  Request.Serialise(result, type);
138  }
139  else
140  {
141  Request.SetStatus(HTTP_BadRequest);
142  Request.SetResponseType(HTTPResponseDefault);
143  }
144 
145  return;
146  }
147 
148  TorcHTTPService::ProcessHTTPRequest(PeerAddress, PeerPort, LocalAddress, LocalPort, Request);
149  return;
150  }
151 
152  // handle options request
153  if (Request.GetHTTPRequestType() == HTTPOptions)
154  {
155  HandleOptions(Request, HTTPHead | HTTPGet | HTTPOptions);
156  return;
157  }
158 
159  Request.SetStatus(HTTP_NotFound);
160  Request.SetResponseType(HTTPResponseDefault);
161 }
