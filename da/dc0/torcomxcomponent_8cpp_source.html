82 void TorcOMXComponent::AnalysePorts(OMX_INDEXTYPE Domain)
83 {
84  OMX_PORT_PARAM_TYPE portparameters;
85  OMX_INITSTRUCTURE(portparameters);
86 
87  if (OMX_GetParameter(m_handle, Domain, &portparameters) != OMX_ErrorNone)
88  {
89  LOG(VB_GENERAL, LOG_ERR, QString("%1: Failed to get port parameters").arg(m_componentName));
90  return;
91  }
92 
93  QString inports;
94  QString outports;
95  int input = 0;
96  int output = 0;
97 
98  for (OMX_U32 port = portparameters.nStartPortNumber; port < (portparameters.nStartPortNumber + portparameters.nPorts); ++port)
99  {
100  OMX_PARAM_PORTDEFINITIONTYPE portdefinition;
101  OMX_INITSTRUCTURE(portdefinition);
102  portdefinition.nPortIndex = port;
103 
104  OMX_ERRORTYPE error = OMX_GetParameter(m_handle, OMX_IndexParamPortDefinition, &portdefinition);
105  if (OMX_ErrorNone == error)
106  {
107  if (OMX_DirInput == portdefinition.eDir)
108  {
109  input++;
110  inports += QString("%1 ").arg(port);
111  m_inputPorts.append(new TorcOMXPort(this, m_handle, port, Domain));
112  }
113  else if (OMX_DirOutput == portdefinition.eDir)
114  {
115  output++;
116  outports += QString("%1 ").arg(port);
117  m_outputPorts.append(new TorcOMXPort(this, m_handle, port, Domain));
118  }
119  }
120  else
121  {
122  OMX_ERROR(error, m_componentName, "Failed to get port definition");
123  }
124  }
125 
126  if (input || output)
127  {
128  LOG(VB_GENERAL, LOG_INFO, QString("%1, %2: %3 input ports (%4), %5 output ports (%6)")
129  .arg(m_componentName).arg(DomainToString(Domain)).arg(input)
130  .arg(inports).arg(output).arg(outports));
131  }
132 }

134 TorcOMXPort* TorcOMXComponent::FindPort(OMX_DIRTYPE Direction, OMX_U32 Index, OMX_INDEXTYPE Domain)
135 {
136  OMX_U32 count = 0;
137  if (OMX_DirInput == Direction && (Index < (OMX_U32)m_inputPorts.size()))
138  {
139  for (int i = 0; i < m_inputPorts.size(); ++i)
140  {
141  if (m_inputPorts.at(i)->GetDomain() == Domain)
142  {
143  if (count == Index)
144  return m_inputPorts.at(i);
145  count++;
146  }
147  }
148  }
149  else if (OMX_DirOutput == Direction && (Index < (OMX_U32)m_outputPorts.size()))
150  {
151  for (int i = 0; i < m_outputPorts.size(); ++i)
152  {
153  if (m_outputPorts.at(i)->GetDomain() == Domain)
154  {
155  if (count == Index)
156  return m_outputPorts.at(i);
157  count++;
158  }
159  }
160  }
161 
162  LOG(VB_GENERAL, LOG_ERR, QString("%1: Failed to find port (Dir: %1, Index %2, Domain %3)")
163  .arg(m_componentName).arg(Index).arg(DomainToString(Domain)));
164  return NULL;
165 }

456 OMX_ERRORTYPE TorcOMXComponent::WaitForResponse(OMX_U32 Command, OMX_U32 Data2, OMX_S32 Timeout)
457 {
458  QElapsedTimer timer;
459  timer.start();
460 
461  LOG(VB_GENERAL, LOG_DEBUG, QString("%1: Waiting for %2 %3")
462  .arg(m_componentName).arg(CommandToString((OMX_COMMANDTYPE)Command)).arg(Data2));
463 
464  while (timer.elapsed() < Timeout)
465  {
466  m_eventQueueLock.lock();
467 
468  QList<TorcOMXEvent>::iterator it = m_eventQueue.begin();
469 
470  for ( ; it != m_eventQueue.end(); ++it)
471  {
472  if ((*it).m_type == OMX_EventCmdComplete && (*it).m_data1 == Command && (*it).m_data2 == Data2)
473  {
474  m_eventQueue.erase(it);
475  m_eventQueueLock.unlock();
476  return OMX_ErrorNone;
477  }
478  else if ((*it).m_type == Command)
479  {
480  LOG(VB_GENERAL, LOG_DEBUG, QString("1 %1 2 %2").arg((*it).m_data1).arg((*it).m_data2));
481  m_eventQueue.erase(it);
482  m_eventQueueLock.unlock();
483  return OMX_ErrorNone;
484  }
485  else if ((*it).m_type == OMX_EventError)
486  {
487  if ((*it).m_data1 == (OMX_U32)OMX_ErrorSameState && (*it).m_data2 == 1)
488  {
489  m_eventQueue.erase(it);
490  m_eventQueueLock.unlock();
491  return OMX_ErrorNone;
492  }
493 
494  LOG(VB_GENERAL, LOG_ERR, QString("%1: Error event '%2' data2 %3")
495  .arg(m_componentName).arg(ErrorToString((OMX_ERRORTYPE)(*it).m_data1)).arg((*it).m_data2));
496  OMX_ERRORTYPE error = (OMX_ERRORTYPE)(*it).m_data1;
497  m_eventQueue.erase(it);
498  m_eventQueueLock.unlock();
499  return error;
500  }
501  }
502 
503  m_eventQueueWait.wait(&m_eventQueueLock, 50);
504  m_eventQueueLock.unlock();
505  }
506 
507  LOG(VB_GENERAL, LOG_INFO, QString("%1: Response never received for command %2 %3")
508  .arg(m_componentName).arg(CommandToString((OMX_COMMANDTYPE)Command)).arg(Data2));
509  return OMX_ErrorMax;
510 }
