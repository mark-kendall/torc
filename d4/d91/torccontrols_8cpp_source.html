43 void TorcControls::Create(const QVariantMap &Details)
44 {
45  QMutexLocker locker(&m_lock);
46 
47  QVariantMap::const_iterator it = Details.constBegin();
48  for( ; it != Details.constEnd(); ++it)
49  {
50  if (it.key() != "controls")
51  continue;
52 
53  QVariantMap control = it.value().toMap();
54  QVariantMap::iterator it2 = control.begin();
55  for ( ; it2 != control.end(); ++it2)
56  {
57  TorcControl::Type type = TorcControl::StringToType(it2.key());
58 
59  // logic, timer or transition
60  if (type == TorcControl::Unknown)
61  {
62  LOG(VB_GENERAL, LOG_ERR, QString("Unknown control type '%1'").arg(it2.key()));
63  continue;
64  }
65 
66  QVariantMap controls = it2.value().toMap();
67  QVariantMap::iterator it3 = controls.begin();
68  for ( ; it3 != controls.end(); ++it3)
69  {
70  QVariantMap details = it3.value().toMap();
71 
72  if (!details.contains("name"))
73  {
74  LOG(VB_GENERAL, LOG_ERR, QString("Ignoring control type '%1' with no <name>").arg(TorcControl::TypeToString(type)));
75  continue;
76  }
77 
78  TorcControl* control = NULL;
79  if (type == TorcControl::Logic)
80  control = new TorcLogicControl(it3.key(), details);
81  else if (type == TorcControl::Timer)
82  control = new TorcTimerControl(it3.key(), details);
83  else
84  control = new TorcTransitionControl(it3.key(), details);
85  controlList.append(control);
86  }
87  }
88  }

100 void TorcControls::Validate(void)
101 {
102  QMutexLocker locker(&m_lock);
103 
104  // We first validate each control.
105 
106  // At this point all sensors, controls and outputs have been parsed successfully and
107  // the objects created.
108  // Validation checks each control has the correct number of valid inputs and outputs and then
109  // connects the appropriate signal/slot pairs. At this point no signals should be triggered.
110 
111  QMutableListIterator<TorcControl*> it(controlList);
112  while (it.hasNext())
113  {
114  TorcControl* control = it.next();
115  if (!control->Validate())
116  {
117  LOG(VB_GENERAL, LOG_ERR, QString("Failed to complete device '%1' - deleting").arg(control->GetUniqueId()));
118  control->DownRef();
119  it.remove();
120  }
121  else
122  {
123  LOG(VB_GENERAL, LOG_INFO, QString("Registered control '%1'").arg(control->Name()));
124  }
125  }
126 
127  LOG(VB_GENERAL, LOG_DEBUG, "Controls validated");
128 
129  // Each control now has a complete list of inputs and outputs so we can now check for circular references.
130  // N.B. Only controls are aware of other devices (i.e. they are created with the expectation that they are
131  // connected to sensors/outputs/other controls. Sensors and outputs are notionally standalone.
132 
133  // Iterate over the controls list and recursively check for circular references.
134  foreach (TorcControl* control, controlList)
135  {
136  QString id = control->GetUniqueId();
137  QString path("");
138  (void)control->CheckForCircularReferences(id, path);
139  }
140 }

159 QVariantMap TorcControls::GetControlList(void)
160 {
161  QVariantMap result;
162  QMutexLocker locker(&m_lock);
163 
164  // iterate over our list for each control type
165  for (int type = TorcControl::Unknown; type < TorcControl::MaxType; type++)
166  {
167  QStringList controlsfortype;
168  foreach (TorcControl *control, controlList)
169  if (control->GetType() == type)
170  controlsfortype.append(control->GetUniqueId());
171 
172  if (!controlsfortype.isEmpty())
173  result.insert(TorcControl::TypeToString(static_cast<TorcControl::Type>(type)), controlsfortype);
174  }
