80 void TorcNetworkInputs::Create(const QVariantMap &Details)
81 {
82  QWriteLocker locker(&m_handlerLock);
83 
84  QVariantMap::const_iterator i = Details.constBegin();
85  for ( ; i != Details.constEnd(); ++i)
86  {
87  // network devices can be <inputs> or <outputs>
88  if ((i.key() != INPUTS_DIRECTORY) && (i.key() != OUTPUTS_DIRECTORY))
89  continue;
90 
91  bool isinput = (i.key() == INPUTS_DIRECTORY);
92 
93  QVariantMap devices = i.value().toMap();
94  QVariantMap::const_iterator j = devices.constBegin();
95  for ( ; j != devices.constEnd(); ++j)
96  {
97  // look for <network> or <constant>
98  QString group = j.key();
99  bool network = group == NETWORK_INPUTS_STRING;
100  bool constant = group == CONSTANT_INPUTS_STRING;
101 
102  if (!network && !constant)
103  continue;
104 
105  QVariantMap details = j.value().toMap();
106  QVariantMap::const_iterator it = details.constBegin();
107  for ( ; it != details.constEnd(); ++it)
108  {
109  // iterate over known types <pwm>, <switch> etc
110  for (int type = TorcInput::Unknown; type != TorcInput::MaxType; type++)
111  {
112  // FIXME using TorcInput::Type to validate both inputs and outputs - will break at some point
113  if (it.key() == TorcCoreUtils::EnumToLowerString<TorcInput::Type>(static_cast<TorcInput::Type>(type)))
114  {
115  QVariantMap input = it.value().toMap();
116  QString defaultvalue = network ? input.value(QStringLiteral("default")).toString() : input.value(QStringLiteral("value")).toString();
117  QString uniqueid = input.value(QStringLiteral("name")).toString(); // ugly... assumes it is present
118 
119  bool ok = false;
120  double defaultdouble = defaultvalue.toDouble(&ok);
121  if (!ok)
122  defaultdouble = 0.0;
123 
124  TorcInput* newinput = nullptr;
125  TorcOutput* newoutput = nullptr;
126  switch (type)
127  {
128  case TorcInput::PWM:
129  {
130  if (isinput)
131  newinput = network ? new TorcNetworkPWMInput(defaultdouble, input) : new TorcPWMInput(defaultdouble, DEVICE_CONSTANT + TorcCoreUtils::EnumToLowerString<TorcInput::Type>(TorcInput::PWM), input);
132  else
133  newoutput = network ? new TorcNetworkPWMOutput(defaultdouble, input) : new TorcPWMOutput(defaultdouble, DEVICE_CONSTANT + TorcCoreUtils::EnumToLowerString<TorcOutput::Type>(TorcOutput::PWM), input);
134  }
135  break;
136  case TorcInput::Switch:
137  {
138  if (isinput)
139  newinput = network ? new TorcNetworkSwitchInput(defaultdouble, input) : new TorcSwitchInput(defaultdouble, DEVICE_CONSTANT + TorcCoreUtils::EnumToLowerString<TorcInput::Type>(TorcInput::Switch), input);
140  else
141  newoutput = network ? new TorcNetworkSwitchOutput(defaultdouble, input) : new TorcSwitchOutput(defaultdouble, DEVICE_CONSTANT + TorcCoreUtils::EnumToLowerString<TorcOutput::Type>(TorcOutput::Switch), input);
142  }
143  break;
144  case TorcInput::Temperature:
145  {
146  if (isinput)
147  newinput = network ? new TorcNetworkTemperatureInput(defaultdouble, input) : new TorcTemperatureInput(defaultdouble, -1000, 1000, DEVICE_CONSTANT + TorcCoreUtils::EnumToLowerString<TorcInput::Type>(TorcInput::Temperature), input);
148  else
149  newoutput = network ? new TorcNetworkTemperatureOutput(defaultdouble, input) : new TorcTemperatureOutput(defaultdouble, DEVICE_CONSTANT + TorcCoreUtils::EnumToLowerString<TorcOutput::Type>(TorcOutput::Temperature), input);
150  }
151  break;
152  case TorcInput::pH:
153  {
154  if (isinput)
155  newinput = network ? new TorcNetworkpHInput(defaultdouble, input) : new TorcpHInput(defaultdouble, DEVICE_CONSTANT + TorcCoreUtils::EnumToLowerString<TorcInput::Type>(TorcInput::pH), input);
156  else
157  newoutput = network ? new TorcNetworkpHOutput(defaultdouble, input) : new TorcpHOutput(defaultdouble, DEVICE_CONSTANT + TorcCoreUtils::EnumToLowerString<TorcOutput::Type>(TorcOutput::pH), input);
158  }
159  break;
160  case TorcInput::Integer:
161  {
162  if (isinput)
163  newinput = network ? new TorcNetworkIntegerInput(defaultdouble, input) : new TorcIntegerInput(defaultdouble, DEVICE_CONSTANT + TorcCoreUtils::EnumToLowerString<TorcInput::Type>(TorcInput::Integer), input);
164  //else
165  // newoutput = network ? new TorcNetworkIntegerOutput(defaultdouble, input) : new TorcIntegerOutput(defaultdouble, DEVICE_CONSTANT + TorcCoreUtils::EnumToLowerString<TorcOutput::Type>(TorcOutput::Integer), input);
166  }
167  break;
168  case TorcInput::Button:
169  {
170  if (constant)
171  {
172  // this should be flagged by the XSD
173  LOG(VB_GENERAL, LOG_ERR, QStringLiteral("Cannot create constant button input"));
174  }
175  else
176  {
177  if (isinput)
178  newinput = new TorcNetworkButtonInput(defaultdouble, input);
179  else
180  {
181  newoutput = new TorcNetworkButtonOutput(defaultdouble, input);
182  }
183  }
184  }
185  default: break;
186  }
187 
188  if (newinput)
189  m_inputs.insert(uniqueid, newinput);
190  if (newoutput)
191  m_outputs.insert(uniqueid, newoutput);
192  }
193  }
194  }
195  }
196  }
