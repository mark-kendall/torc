82 void TorcXmlValidator::Validate(void)
83 {
84  xmlSetStructuredErrorFunc(NULL, NULL);
85  xmlSetGenericErrorFunc(this, HandleMessage);
86  xmlThrDefSetStructuredErrorFunc(NULL, NULL);
87  xmlThrDefSetGenericErrorFunc(this, HandleMessage);
88 
89  xmlSchemaParserCtxtPtr parsercontext = NULL;
90 
91  if (!m_xsdFile.isEmpty())
92  {
93  parsercontext = xmlSchemaNewParserCtxt(m_xsdFile.toLocal8Bit().constData());
94  }
95  else if (!m_xsdData.isEmpty())
96  {
97  parsercontext = xmlSchemaNewMemParserCtxt(m_xsdData.constData(), m_xsdData.size());
98  }
99 
100  if (!parsercontext)
101  {
102  LOG(VB_GENERAL, LOG_ERR, "Failed to create parser context for XSD");
103  return;
104  }
105 
106  xmlSchemaPtr schema = xmlSchemaParse(parsercontext);
107 
108  if (!schema)
109  {
110  xmlSchemaFreeParserCtxt(parsercontext);
111  LOG(VB_GENERAL, LOG_ERR, "Failed to parse XSD");
112  return;
113  }
114 
115  xmlSchemaValidCtxtPtr validcontext = xmlSchemaNewValidCtxt(schema);
116 
117  if (!validcontext)
118  {
119  xmlSchemaFree(schema);
120  xmlSchemaFreeParserCtxt(parsercontext);
121  LOG(VB_GENERAL, LOG_ERR, "Could not create XSD schema validation context");
122  return;
123  }
124 
125  if (!m_silent)
126  LOG(VB_GENERAL, LOG_INFO, "XSD loaded and validated");
127 
128  m_xsdDone = true;
129 
130  xmlDocPtr xmldocumentptr = xmlParseFile(m_xmlFile.toLocal8Bit().constData());
131  int result = xmlSchemaValidateDoc(validcontext, xmldocumentptr);
132 
133  if (result == 0)
134  m_valid = true;
135 
136  xmlFreeDoc(xmldocumentptr);
137  xmlSchemaFreeValidCtxt(validcontext);
138  xmlSchemaFree(schema);
139  xmlSchemaFreeParserCtxt(parsercontext);
140 }
