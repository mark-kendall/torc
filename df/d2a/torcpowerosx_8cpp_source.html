44 TorcPowerOSX::TorcPowerOSX()
45  : m_powerRef(NULL),
46  m_rootPowerDomain(0),
47  m_powerNotifier(MACH_PORT_NULL),
48  m_powerNotifyPort(NULL)
49 {
50  CocoaAutoReleasePool pool;
51 
52  if (!gAdminRunLoop)
53  {
54  LOG(VB_GENERAL, LOG_ERR, "OS X callback run loop not present - aborting");
55  return;
56  }
57 
58  // Register for power status updates
59  m_rootPowerDomain = IORegisterForSystemPower(this, &m_powerNotifyPort, PowerCallBack, &m_powerNotifier);
60  if (m_rootPowerDomain)
61  {
62  CFRunLoopAddSource(gAdminRunLoop,
63  IONotificationPortGetRunLoopSource(m_powerNotifyPort),
64  kCFRunLoopDefaultMode);
65  }
66  else
67  {
68  LOG(VB_GENERAL, LOG_ERR, "Failed to setup power status callback");
69  }
70 
71  // Is there a battery?
72  CFArrayRef batteryinfo = NULL;
73 
74  if (IOPMCopyBatteryInfo(kIOMasterPortDefault, &batteryinfo) == kIOReturnSuccess)
75  {
76  CFRelease(batteryinfo);
77 
78  // register for notification of power source changes
79  m_powerRef = IOPSNotificationCreateRunLoopSource(PowerSourceCallBack, this);
80  if (m_powerRef)
81  {
82  CFRunLoopAddSource(gAdminRunLoop, m_powerRef, kCFRunLoopDefaultMode);
83  Refresh();
84  }
85  }
86 
87  if (!m_powerRef)
88  LOG(VB_GENERAL, LOG_ERR, "Failed to setup power source callback");
89 
90  // Set capabilities
91  m_canShutdown->SetValue(QVariant((bool)true));
92  m_canSuspend->SetValue(QVariant((bool)IOPMSleepEnabled()));
93  m_canHibernate->SetValue(QVariant((bool)true));
94  m_canRestart->SetValue(QVariant((bool)true));
95 
96  Debug();
97 }

179 void TorcPowerOSX::Refresh(void)
180 {
181  if (!m_powerRef)
182  return;
183 
184  CFTypeRef info = IOPSCopyPowerSourcesInfo();
185  CFArrayRef list = IOPSCopyPowerSourcesList(info);
186 
187  for (int i = 0; i < CFArrayGetCount(list); i++)
188  {
189  CFTypeRef source = CFArrayGetValueAtIndex(list, i);
190  CFDictionaryRef description = IOPSGetPowerSourceDescription(info, source);
191 
192  if ((CFBooleanRef)CFDictionaryGetValue(description, CFSTR(kIOPSIsPresentKey)) == kCFBooleanFalse)
193  continue;
194 
195  CFStringRef type = (CFStringRef)CFDictionaryGetValue(description, CFSTR(kIOPSTransportTypeKey));
196  if (type && CFStringCompare(type, CFSTR(kIOPSInternalType), 0) == kCFCompareEqualTo)
197  {
198  CFStringRef state = (CFStringRef)CFDictionaryGetValue(description, CFSTR(kIOPSPowerSourceStateKey));
199  if (state && CFStringCompare(state, CFSTR(kIOPSACPowerValue), 0) == kCFCompareEqualTo)
200  {
201  m_batteryLevel = TorcPower::ACPower;
202  }
203  else if (state && CFStringCompare(state, CFSTR(kIOPSBatteryPowerValue), 0) == kCFCompareEqualTo)
204  {
205  int32_t current;
206  int32_t max;
207  CFNumberRef capacity = (CFNumberRef)CFDictionaryGetValue(description, CFSTR(kIOPSCurrentCapacityKey));
208  CFNumberGetValue(capacity, kCFNumberSInt32Type, &current);
209  capacity = (CFNumberRef)CFDictionaryGetValue(description, CFSTR(kIOPSMaxCapacityKey));
210  CFNumberGetValue(capacity, kCFNumberSInt32Type, &max);
211  m_batteryLevel = (int)(((qreal)current / ((qreal)max)) * 100.0);
212  }
213  else
214  {
215  m_batteryLevel = TorcPower::UnknownPower;
216  }
217  }
218  }
219 
220  CFRelease(list);
221  CFRelease(info);
222 
223  BatteryUpdated(m_batteryLevel);
224 }

233 void TorcPowerOSX::PowerCallBack(void *Reference, io_service_t Service,
234  natural_t Type, void *Data)
235 {
236  (void)Service;
237 
238  CocoaAutoReleasePool pool;
239  TorcPowerOSX* power = static_cast<TorcPowerOSX*>(Reference);
240 
241  if (power)
242  {
243  switch (Type)
244  {
245  case kIOMessageCanSystemPowerOff:
246  IOAllowPowerChange(power->m_rootPowerDomain, (long)Data);
247  power->ShuttingDown();
248  break;
249  case kIOMessageCanSystemSleep:
250  IOAllowPowerChange(power->m_rootPowerDomain, (long)Data);
251  power->Suspending();
252  break;
253  case kIOMessageSystemWillPowerOff:
254  IOAllowPowerChange(power->m_rootPowerDomain, (long)Data);
255  power->ShuttingDown();
256  break;
257  case kIOMessageSystemWillRestart:
258  IOAllowPowerChange(power->m_rootPowerDomain, (long)Data);
259  power->Restarting();
260  break;
261  case kIOMessageSystemWillSleep:
262  IOAllowPowerChange(power->m_rootPowerDomain, (long)Data);
263  power->Suspending();
264  break;
265  case kIOMessageSystemHasPoweredOn:
266  power->WokeUp();
267  break;
268  }
269  }
270 }

288 OSStatus SendAppleEventToSystemProcess(AEEventID EventToSend)
289 {
290  AEAddressDesc targetDesc;
291  static const ProcessSerialNumber kPSNOfSystemProcess = { 0, kSystemProcess };
292  AppleEvent eventReply = { typeNull, NULL };
293  AppleEvent appleEventToSend = { typeNull, NULL };
294 
295  OSStatus error = AECreateDesc(typeProcessSerialNumber, &kPSNOfSystemProcess,
296  sizeof(kPSNOfSystemProcess), &targetDesc);
297 
298  if (error != noErr)
299  return error;
300 
301  error = AECreateAppleEvent(kCoreEventClass, EventToSend, &targetDesc,
302  kAutoGenerateReturnID, kAnyTransactionID, &appleEventToSend);
303 
304  AEDisposeDesc(&targetDesc);
305  if (error != noErr)
306  return error;
307 
308  error = AESendMessage(&appleEventToSend, &eventReply, kAENormalPriority, kAEDefaultTimeout);
309 
310  AEDisposeDesc(&appleEventToSend);
311 
312  if (error != noErr)
313  return error;
314 
315  AEDisposeDesc(&eventReply);
316 
317  return error;
318 }
