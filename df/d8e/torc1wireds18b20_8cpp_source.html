44 void Torc1WireReadThread::Start(void)
45 {
46  // create and setup timer in correct thread
47  m_timer = new QTimer();
48  m_timer->setTimerType(Qt::CoarseTimer);
49 
50  // restart the timer each time to handle periods of load/delay
51  m_timer->setSingleShot(true);
52 
53  // 'this' is in the parent's thread and connect will by default use an indirect connection,
54  // so force a direct connection to ensure Read operates in the 1Wire thread.
55  // This way we don't need an extra object created in the read thread to handle one function...
56  connect(m_timer, SIGNAL(timeout()), this, SLOT(Read()), Qt::DirectConnection);
57 
58  // randomise start time for each input
59  m_timer->start(qrand() % MIN_READ_INTERVAL);
60 }

68 void Torc1WireReadThread::Read(void)
69 {
70  QFile file(m_file);
71 
72  // open
73  if (!file.open(QIODevice::ReadOnly))
74  {
75  LOG(VB_GENERAL, LOG_ERR, QString("Failed to read device '%1' (Error #%2: '%3')")
76  .arg(m_file).arg(file.error()).arg(file.errorString()));
77  }
78  else
79  {
80  // read
81  QTextStream text(&file);
82 
83  // check crc
84  QString line = text.readLine();
85  if (!line.contains("crc") || !line.contains("YES"))
86  {
87  LOG(VB_GENERAL, LOG_ERR, QString("CRC check failed for device %1").arg(m_file));
88  }
89  else
90  {
91  // read temp
92  line = text.readLine().trimmed();
93  int index = line.lastIndexOf("t=");
94  if (index < 0)
95  {
96  LOG(VB_GENERAL, LOG_ERR, QString("Failed to parse temperature for device %1").arg(m_file));
97  }
98  else
99  {
100  line = line.mid(index + 2);
101  bool ok = false;
102  double temp = line.toDouble(&ok);
103  if (ok)
104  {
105  file.close();
106  m_parent->Read(temp / 1000.0, true);
107  m_timer->start(MIN_READ_INTERVAL);
108  return;
109  }
110 
111  LOG(VB_GENERAL, LOG_ERR, QString("Failed to convert temperature for device %1").arg(m_file));
112  }
113  }
114  }
115 
116  file.close();
117  m_parent->Read(0, false);
118 
119  m_timer->start(MIN_READ_INTERVAL);
120 }
