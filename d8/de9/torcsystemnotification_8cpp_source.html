58 TorcSystemNotification::TorcSystemNotification(const QVariantMap &Details)
59  : TorcNotification(Details),
60  m_events()
61 {
62  if (uniqueId.isEmpty() || m_notifierNames.isEmpty() || m_body.isEmpty())
63  return;
64 
65  // NB event names are also validated in the xsd
66  if (Details.contains("inputs"))
67  {
68  QVariantMap inputs = Details.value("inputs").toMap();
69  QVariantMap::const_iterator it = inputs.constBegin();
70  for ( ; it != inputs.constEnd(); ++it)
71  {
72  if (it.key() == "event")
73  {
74  int event = Torc::StringToAction(it.value().toString().trimmed());
75  if (event != -1)
76  m_events.append((Torc::Actions)event);
77  }
78  }
79  }
80 
81  if (m_events.isEmpty())
82  {
83  LOG(VB_GENERAL, LOG_ERR, QString("System notification '%1' has no events to listen for").arg(uniqueId));
84  }
85  else
86  {
87  gLocalContext->AddObserver(this);

116 bool TorcSystemNotification::event(QEvent *Event)
117 {
118  if (Event && (Event->type() == TorcEvent::TorcEventType))
119  {
120  TorcEvent* torcevent = dynamic_cast<TorcEvent*>(Event);
121  if (torcevent)
122  {
123  QMutexLocker locker(&lock);
124  Torc::Actions event = (Torc::Actions)torcevent->GetEvent();
125  if (m_events.contains(event))
126  {
127  QMap<QString,QString> custom;
128  custom.insert("event", Torc::ActionToString(event));
129  QVariantMap message = TorcNotify::gNotify->SetNotificationText(m_title, m_body, custom);
130  emit Notify(message);
131  }
132  }
133  }
134 
135  return TorcNotification::event(Event);
136 }
