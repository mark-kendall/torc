108 void TorcHTTPRequest::Initialise(const QString &Method)
109 {
110  QStringList items = Method.split(gRegExp, QString::SkipEmptyParts);
111  QString item;
112 
113  if (!items.isEmpty())
114  {
115  item = items.takeFirst();
116 
117  // response of type 'HTTP/1.1 200 OK'
118  if (item.startsWith("HTTP"))
119  {
120  m_type = HTTPResponse;
121 
122  // HTTP/1.1
123  m_protocol = ProtocolFromString(item.trimmed());
124 
125  // 200 OK
126  if (!items.isEmpty())
127  m_responseStatus = StatusFromString(items.takeFirst().trimmed());
128  }
129  // request of type 'GET /method HTTP/1.1'
130  else
131  {
132  m_type = HTTPRequest;
133 
134  // GET
135  m_requestType = RequestTypeFromString(item.trimmed());
136 
137  if (!items.isEmpty())
138  {
139  // /method
140  QUrl url = QUrl::fromEncoded(items.takeFirst().toUtf8());
141  m_path = url.path();
142  m_fullUrl = url.toString();
143 
144  int index = m_path.lastIndexOf("/");
145  if (index > -1)
146  {
147  m_method = m_path.mid(index + 1).trimmed();
148  m_path = m_path.left(index + 1).trimmed();
149  }
150 
151  if (url.hasQuery())
152  {
153  QStringList pairs = url.query().split('&');
154  foreach (QString pair, pairs)
155  {
156  int index = pair.indexOf('=');
157  QString key = pair.left(index);
158  QString val = pair.mid(index + 1);
159  m_queries.insert(key, val);
160  }
161  }
162  }
163 
164  // HTTP/1.1
165  if (!items.isEmpty())
166  m_protocol = ProtocolFromString(items.takeFirst());
167  }
168  }
169 
170  if (m_protocol > HTTPOneDotZero)
171  m_connection = HTTPConnectionKeepAlive;
172 
173  QString connection = m_headers.value("Connection").toLower();
174 
175  if (connection == "keep-alive")
176  m_connection = HTTPConnectionKeepAlive;
177  else if (connection == "close")
178  m_connection = HTTPConnectionClose;
179 
180  LOG(VB_GENERAL, LOG_DEBUG, QString("HTTP request: path '%1' method '%2'").arg(m_path).arg(m_method));
181 }

386  }

392  QScopedPointer<QByteArray> headers(new QByteArray);
393  QTextStream response(headers.data());
394

396  response << "Date: " << QDateTime::currentDateTimeUtc().toString(DateFormat) << "\r\n";
397  response << "Server: " << TorcHTTPServer::PlatformName() << "\r\n";
398  response << "Connection: " << TorcHTTPRequest::ConnectionToString(m_connection) << "\r\n";
399  response << "Accept-Ranges: bytes\r\n";
400 
401  if (m_cache & HTTPCacheNone)
402  {
403  response << "Cache-Control: private, no-cache, no-store, must-revalidate\r\nExpires: 0\r\nPragma: no-cache\r\n";
404  }
405  else
406  {
407  // cache-control (in preference to expires for its simplicity)
408  if (m_cache & HTTPCacheShortLife)
409  response << "Cache-Control: public, max-age=3600\r\n"; // 1 hour
410  else if (m_cache & HTTPCacheLongLife)
411  response << "Cache-Control: public, max-age=31536000\r\n"; // 1 year (max per spec)
412 
413  // either last-modified or etag (not both) if requested
414  if (!m_cacheTag.isEmpty())
415  {
416  if (m_cache & HTTPCacheETag)
417  response << QString("ETag: \"%1\"\r\n").arg(m_cacheTag);
418  else if (m_cache & HTTPCacheLastModified)
419  response << QString("Last-Modified: %1\r\n").arg(m_cacheTag);
420  }

424  // - it was requested by the client.
425  // - zlip support is available locally.

427  // - there is some content and it is smaller than 1Mb in size (arbitrary limit)
428  // - the response is not a range request with single or multipart response

430  m_headers.contains("Accept-Encoding") && m_headers.value("Accept-Encoding").contains("gzip", Qt::CaseInsensitive))
431  {

436  }

441  }

443  response << "Content-Encoding: gzip\r\n";
444  }
445 
446  if (multipart)
447  response << "Content-Type: multipart/byteranges; boundary=STaRT\r\n";
448  else if (m_responseType != HTTPResponseNone)
449  response << contentheader;
450 
451  if (m_allowed)

453  response << "Content-Length: " << QString::number(sendsize) << "\r\n";
454

457  else if (m_responseStatus == HTTP_RequestedRangeNotSatisfiable)
458  response << "Content-Range: bytes */" << QString::number(totalsize) << "\r\n";
459 
460  if (m_responseStatus == HTTP_MovedPermanently)
461  response << "Location: " << m_redirectedTo.toLatin1() << "\r\n";

466  response << it.key().toLatin1() << ": " << it.value().toLatin1() << "\r\n";

919 QList<QPair<quint64,quint64> > TorcHTTPRequest::StringToRanges(const QString &Ranges, qint64 Size, qint64 &SizeToSend)
920 {
921  qint64 tosend = 0;
922  QList<QPair<quint64,quint64> > results;
923  if (Size < 1)
924  return results;
925 
926  if (Ranges.contains("bytes", Qt::CaseInsensitive))
927  {
928  QStringList newranges = Ranges.split("=");
929  if (newranges.size() == 2)
930  {
931  QStringList rangelist = newranges[1].split(",", QString::SkipEmptyParts);
932 
933  foreach (QString range, rangelist)
934  {
935  QStringList parts = range.split("-");
936 
937  if (parts.size() != 2)
938  continue;
939 
940  qint64 start = 0;
941  qint64 end = 0;
942  bool ok = false;
943  bool havefirst = !parts[0].trimmed().isEmpty();
944  bool havesecond = !parts[1].trimmed().isEmpty();
945 
946  if (havefirst && havesecond)
947  {
948  start = parts[0].toULongLong(&ok);
949  if (ok)
950  {
951  // invalid per spec
952  if (start >= Size)
953  continue;
954 
955  end = parts[1].toULongLong(&ok);
956 
957  // invalid per spec
958  if (end < start)
959  continue;
960  }
961  }
962  else if (havefirst && !havesecond)
963  {
964  start = parts[0].toULongLong(&ok);
965 
966  // invalid per spec
967  if (ok && start >= Size)
968  continue;
969 
970  end = Size - 1;
971  }
972  else if (!havefirst && havesecond)
973  {
974  end = Size -1;
975  start = parts[1].toULongLong(&ok);
976  if (ok)
977  {
978  start = Size - start;
979 
980  // invalid per spec
981  if (start >= Size)
982  continue;
983  }
984  }
985 
986  if (ok)
987  {
988  results << QPair<quint64,quint64>(start, end);
989  tosend += end - start + 1;
990  }
991  }
992  }
993  }
994 
995  if (results.isEmpty())
996  {
997  SizeToSend = 0;
998  LOG(VB_GENERAL, LOG_ERR, QString("Error parsing byte ranges ('%1')").arg(Ranges));
999  }
1000  else
1001  {
1002  // rationalise overlapping range requests
1003  bool check = true;
1004 
1005  while (results.size() > 1 && check)
1006  {
1007  check = false;
1008  quint64 laststart = results[0].first;
1009  quint64 lastend = results[0].second;
1010 
1011  for (int i = 1; i < results.size(); ++i)
1012  {
1013  quint64 thisstart = results[i].first;
1014  quint64 thisend = results[i].second;
1015 
1016  if (thisstart > lastend)
1017  {
1018  laststart = thisstart;
1019  lastend = thisend;
1020  continue;
1021  }
1022 
1023  tosend -= (lastend - laststart + 1);
1024  tosend -= (thisend - thisstart + 1);
1025  results[i - 1].first = qMin(laststart, thisstart);
1026  results[i - 1].second = qMax(lastend, thisend);
1027  tosend += results[i - 1].second - results[i - 1].first + 1;
1028  results.removeAt(i);
1029  check = true;
1030  break;
1031  }
1032  }
1033 
1034  // set content size
1035  SizeToSend = tosend;
1036  }
1037 
1038  return results;
1039 }
