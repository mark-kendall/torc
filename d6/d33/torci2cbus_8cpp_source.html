67 void TorcI2CBus::Create(const QVariantMap &Details)
68 {
69  QMutexLocker locker(&m_lock);
70 
71  QVariantMap::const_iterator i = Details.constBegin();
72  for ( ; i != Details.constEnd(); ++i)
73  {
74  // I2C devices can be <sensors> or <outputs> (and both at the same time)
75  if (i.key() != INPUTS_DIRECTORY && i.key() != OUTPUTS_DIRECTORY)
76  continue;
77 
78  QVariantMap i2c = i.value().toMap();
79  QVariantMap::iterator it = i2c.begin();
80  for ( ; it != i2c.end(); ++it)
81  {
82  // look for <i2c>
83  if (it.key() != I2C)
84  continue;
85 
86  QVariantMap devices = it.value().toMap();
87  QVariantMap::iterator it2 = devices.begin();
88  for ( ; it2 != devices.end(); ++it2)
89  {
90  QVariantMap details = it2.value().toMap();
91 
92  // device needs an address
93  if (!details.contains("i2caddress"))
94  {
95  LOG(VB_GENERAL, LOG_ERR, QString("I2C device '%1' needs <i2caddress>").arg(it2.key()));
96  continue;
97  }
98 
99  bool ok = false;
100  // N.B. assumes hexadecimal 0xXX
101  int address = details.value("i2caddress").toString().toInt(&ok, 16);
102  if (!ok)
103  {
104  LOG(VB_GENERAL, LOG_ERR, QString("Failed to parse I2C address from '%1' for device '%2'")
105  .arg(details.value("address").toString()).arg(it2.key()));
106  continue;
107  }
108 
109  // TODO - handle defining the same device in sensors and outputs
110  if (m_devices.contains(address))
111  {
112  LOG(VB_GENERAL, LOG_ERR, QString("I2C bus already contains device at 0x%1 - ignoring").arg(address, 0, 16));
113  continue;
114  }
115 
116  TorcI2CDevice* device = NULL;
117  TorcI2CDeviceFactory* factory = TorcI2CDeviceFactory::GetTorcI2CDeviceFactory();
118  for ( ; factory; factory = factory->NextFactory())
119  {
120  device = factory->Create(address, it2.key(), details);
121  if (device)
122  break;
123  }
124 
125  if (device)
126  m_devices.insert(address, device);
127  else
128  LOG(VB_GENERAL, LOG_ERR, QString("Unable to find handler for I2C device '%1'").arg(it2.key()));
129  }
130  }
131  }
132 }
