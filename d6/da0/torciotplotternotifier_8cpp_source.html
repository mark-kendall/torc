58 TorcNetworkRequest* TorcIoTPlotterNotifier::CreateRequest(void)
59 {
60  QUrl url(IOTPLOTTER_UPDATE_URL + m_feedId);
61  QNetworkRequest qrequest(url);
62  qrequest.setHeader(QNetworkRequest::ContentTypeHeader, "application/x-www-form-urlencoded");
63  qrequest.setRawHeader("api-key", m_apiKey.toLocal8Bit());
64 
65  QJsonObject data;
66  QJsonObject raw;
67  // NB from Qt 5.8 we can use currentSecsSinceEpoch
68  QString epoch = QString("%1").arg(((quint64)QDateTime::currentMSecsSinceEpoch() / 1000));
69 
70  for (int i = 0; i < m_maxFields; i++)
71  {
72  if (!m_fieldValues[i].isEmpty())
73  {
74  if (m_reverseFields.contains(i))
75  {
76  QJsonObject pair;
77  pair.insert("value", 55);//m_fieldValues[i].toLongLong());
78  //pair.insert("epoch", epoch.toLongLong());
79  QJsonArray values;
80  values.append(pair);
81  raw.insert(m_reverseFields[i].toLocal8Bit(), values);
82  }
83  }
84 
85  // reset the fields
86  m_fieldValues[i] = QString("");
87  }
88 
89  data.insert("data", raw);
90  QJsonDocument doc(data);
91  QByteArray serialiseddata = doc.toJson(QJsonDocument::Indented);
92  qrequest.setHeader(QNetworkRequest::ContentLengthHeader, serialiseddata.size());
93  return new TorcNetworkRequest(qrequest, serialiseddata, &m_networkAbort);
94 }
