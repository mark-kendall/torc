51 {

57  // handle options request

63

85 void TorcHTMLStaticContent::GetJavascriptConfiguration(TorcHTTPRequest &Request)
86 {
87  // populate the list of static constants and translations
88  QVariantMap strings = TorcStringFactory::GetTorcStrings();
89 
90  // generate dynamic variables
91  strings.insert("ServicesPath", TORC_SERVICES_DIR);
92  strings.insert("UserService", TORC_USER_SERVICE);
93  strings.insert("SettingsPath", TORC_SETTINGS_DIR);
94  strings.insert("RootSetting", TORC_ROOT_SETTING);
95  strings.insert("TorcRealm", TORC_REALM);
96  strings.insert("PortSetting", TORC_PORT_SERVICE);
97  strings.insert("SecureSetting",TORC_SSL_SERVICE);
98  strings.insert("TorcConfFile", TORC_CONTENT_DIR + TORC_CONFIG_FILE);
99 
100  // and generate javascript
101  QJsonObject json = QJsonObject::fromVariantMap(strings);
102  QByteArray result("var torc = ");
103  result.append(QJsonDocument(json).toJson());
104  if (result.endsWith("\n"))
105  result.chop(1);
106  result.append(";\r\n\r\nif (Object.freeze) { Object.freeze(torc); }\r\n");
107 
108  Request.SetStatus(HTTP_OK);
109  Request.SetResponseType(HTTPResponseJSONJavascript);
110  Request.SetResponseContent(result);
111 }
