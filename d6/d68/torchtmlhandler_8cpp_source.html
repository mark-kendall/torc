65 {

71  // handle options request

74  // this is the 'global' options - return everything possible

76  HandleOptions(Request, HTTPHead | HTTPGet | HTTPPost | HTTPPut | HTTPDelete | HTTPOptions);
77  else
78  HandleOptions(Request, HTTPHead | HTTPGet | HTTPOptions);
79  return;

81 
82  // allowed file?

85  {

88 
89  HandleFile(Request, m_pathToContent + url, HTTPCacheNone);
90  }
91  // I think this is unused/unreachable. img directory is served by TorcHTMLStaticContent
92  else if (url.endsWith(".png", Qt::CaseInsensitive) || url.endsWith(".ico", Qt::CaseInsensitive))
93  {
94  HandleFile(Request, m_pathToContent + "/img" + url, HTTPCacheLongLife | HTTPCacheLastModified);
95  }
96  else
