97 int TorcXSDTest::RunXSDTestSuite(TorcCommandLine *CommandLine)
98 {
99 #if defined(USING_XMLPATTERNS) || defined(USING_LIBXML2)
100  if (!CommandLine)
101  return TORC_EXIT_INVALID_CMDLINE;
102 
103  if (int error = TorcLocalContext::Create(CommandLine, false))
104  return error;
105 
106  QString directory = CommandLine->GetValue("xsdtest").toString();
107  LOG(VB_GENERAL, LOG_INFO, QString("Starting XSD test from %1").arg(directory));
108 
109  QString basexsd = GetTorcShareDir() + "/html/torc.xsd";
110  QByteArray fullxsd = TorcCentral::GetCustomisedXSD(basexsd);
111  if (fullxsd.isEmpty())
112  {
113  TorcLocalContext::TearDown();
114  return TORC_EXIT_UNKOWN_ERROR;
115  }
116 
117  QDir dir(directory);
118  QStringList filters;
119  filters << "*.xml";
120 
121  QStringList testfiles = dir.entryList(filters, QDir::Files | QDir::NoDotAndDotDot | QDir::Readable | QDir::CaseSensitive, QDir::Name);
122 
123  if (testfiles.isEmpty())
124  {
125  LOG(VB_GENERAL, LOG_INFO, QString("Failed to find any xml files in %1").arg(dir.path()));
126  TorcLocalContext::TearDown();
127  return TORC_EXIT_UNKOWN_ERROR;
128  }
129 
130  LOG(VB_GENERAL, LOG_INFO, QString("Found %1 files.").arg(testfiles.size()));
131  int passcount = 0;
132  foreach (QString file, testfiles)
133  {
134  // these should all FAIL!
135  QString path = directory + "/" + file;
136  LOG(VB_GENERAL, LOG_INFO, path);
137  TorcXmlValidator validator(path, fullxsd, true);
138  if (validator.Validated())
139  {
140  LOG(VB_GENERAL, LOG_ERR, QString("Unexpected pass: %1").arg(path));
141  passcount++;
142  }
143  }
144 
145  if (!passcount)
146  LOG(VB_GENERAL, LOG_INFO, "All test files failed as expected");
147  else
148  LOG(VB_GENERAL, LOG_ERR, QString("%1 unexpected passes.").arg(passcount));
149 
150  TorcLocalContext::TearDown();
151 
152 #else
153  (void) CommandLine;
154  LOG(VB_GENERAL, LOG_INFO, "XML validation not available.");
155 #endif
156 
157  return TORC_EXIT_OK;
158 }
