60 void TorcPListSerialiser::PListFromVariant(const QString &Name, const QVariant &Value, bool NeedKey)
61 {
62  if (Value.isNull())
63  {
64  if (NeedKey)
65  m_xmlStream.writeTextElement("key", Name);
66  m_xmlStream.writeEmptyElement("null");
67  return;
68  }
69 
70  switch ((int)Value.type())
71  {
72  case QMetaType::QVariantList: PListFromList(Name, Value.toList()); break;
73  case QMetaType::QStringList: PListFromStringList(Name, Value.toStringList()); break;
74  case QMetaType::QVariantMap: PListFromMap(Name, Value.toMap()); break;
75  case QMetaType::QDateTime:
76  {
77  if (Value.toDateTime().isValid())
78  {
79  if (NeedKey)
80  m_xmlStream.writeTextElement("key", Name);
81  m_xmlStream.writeTextElement("date", Value.toDateTime().toUTC().toString("yyyy-MM-ddThh:mm:ssZ"));
82  }
83  break;
84  }
85  case QMetaType::QByteArray:
86  {
87  if (!Value.toByteArray().isNull())
88  {
89  if (NeedKey)
90  m_xmlStream.writeTextElement("key", Name);
91  m_xmlStream.writeTextElement("data", Value.toByteArray().toBase64().data());
92  }
93  break;
94  }
95  case QMetaType::Bool:
96  {
97  if (NeedKey)
98  m_xmlStream.writeTextElement("key", Name);
99  m_xmlStream.writeEmptyElement(Value.toBool() ? "true" : "false");
100  break;
101  }
102  case QMetaType::Char:
103  {
104  if (NeedKey)
105  m_xmlStream.writeTextElement("key", Name);
106  m_xmlStream.writeEmptyElement("fill");
107  break;
108  }
109  break;
110  case QMetaType::UInt:
111  case QMetaType::UShort:
112  case QMetaType::ULong:
113  case QMetaType::ULongLong:
114  {
115  if (NeedKey)
116  m_xmlStream.writeTextElement("key", Name);
117  m_xmlStream.writeTextElement("integer", QString::number(Value.toULongLong()));
118  break;
119  }
120 
121  case QMetaType::Int:
122  case QMetaType::Short:
123  case QMetaType::Long:
124  case QMetaType::LongLong:
125  case QMetaType::Float:
126  case QMetaType::Double:
127  {
128  if (NeedKey)
129  m_xmlStream.writeTextElement("key", Name);
130  m_xmlStream.writeTextElement("real", QString("%1").arg(Value.toDouble(), 0, 'f', 6));
131  break;
132  }
133 
134  case QMetaType::QString:
135  default:
136  {
137  if (NeedKey)
138  m_xmlStream.writeTextElement("key", Name);
139  m_xmlStream.writeTextElement("string", Value.toString());
140  break;
141  }
142  }
143 }

145 void TorcPListSerialiser::PListFromList(const QString &Name, const QVariantList &Value)
146 {
147  if (!Value.isEmpty())
148  {
149  int type = Value[0].type();
150 
151  QVariantList::const_iterator it = Value.begin();
152  for ( ; it != Value.end(); ++it)
153  {
154  if ((int)(*it).type() != type)
155  {
156  PListFromVariant("Error", QVARIANT_ERROR);
157  return;
158  }
159  }
160  }
161 
162  m_xmlStream.writeTextElement("key", Name);
163  m_xmlStream.writeStartElement("array");
164 
165  QVariantList::const_iterator it = Value.begin();
166  for ( ; it != Value.end(); ++it)
167  PListFromVariant(Name, (*it), false);
168 
169  m_xmlStream.writeEndElement();
170 }
