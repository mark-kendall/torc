78 void TorcXMLSerialiser::VariantToXML(const QString &Name, const QVariant &Value)
79 {
80  switch(static_cast<QMetaType::Type>(Value.type()))
81  {
82  case QMetaType::QVariantList: ListToXML(Name, Value.toList()); return;
83  case QMetaType::QStringList: StringListToXML(Name, Value.toStringList()); return;
84  case QMetaType::QVariantMap: MapToXML(Name, Value.toMap()); return;
85  case (QVariant::Type)QMetaType::QDateTime:
86  {
87  QDateTime datetime(Value.toDateTime());
88  if (datetime.isNull())
89  m_xmlStream.writeAttribute("xsi:nil", "true");
90  m_xmlStream.writeCharacters(datetime.toString(Qt::ISODate));
91  return;
92  }
93  default: m_xmlStream.writeCharacters(Value.toString()); return;
94  }
95 }

97 void TorcXMLSerialiser::ListToXML(const QString &Name, const QVariantList &Value)
98 {
99  if (!Value.isEmpty())

102  QVariantList::const_iterator it = Value.begin();
