45 void TorcOutputs::Graph(QByteArray* Data)
46 {
47  if (!Data)
48  return;
49 
50  QString start =
51  QString(" subgraph cluster_2 {\r\n"
52  " label = \"%1\";\r\n"
53  " style=filled;\r\n"
54  " fillcolor=\"grey95\";\r\n").arg(tr("Outputs"));
55  Data->append(start);
56 
57  foreach(TorcOutput* output, outputList)
58  {
59  QString id = output->GetUniqueId();
60  QString label = output->GetUserName();
61  QString url = output->GetPresentationURL();
62  QString desc;
63  QStringList source = output->GetDescription();
64  foreach (QString item, source)
65  if (!item.isEmpty())
66  desc.append(QString(DEVICE_LINE_ITEM).arg(item));
67  desc.append(QString(DEVICE_LINE_ITEM).arg(tr("Default %1").arg(output->GetDefaultValue())));
68  desc.append(QString(DEVICE_LINE_ITEM).arg(tr("Valid %1").arg(output->GetValid())));
69  desc.append(QString(DEVICE_LINE_ITEM).arg(tr("Value %1").arg(output->GetValue())));
70 
71  if (label.isEmpty())
72  label = id;
73  QString link = url.isEmpty() ? QString() : QString(" href=\"%1\"").arg(url);
74  Data->append(QString(" \"%1\" [shape=record id=\"%1\" label=<<B>%2</B>%3>%4];\r\n")
75  .arg(id).arg(label).arg(desc).arg(link));
76  }
77 
78  Data->append(" }\r\n\r\n");
79 }

88 QVariantMap TorcOutputs::GetOutputList(void)
89 {
90  QVariantMap result;
91  QReadLocker locker(&m_httpServiceLock);
92 
93  // iterate over our list for each output type
94  for (int type = TorcOutput::Unknown; type < TorcOutput::MaxType; type++)
95  {
96  QStringList outputsfortype;
97  foreach (TorcOutput *output, outputList)
98  if (output->GetType() == type)
99  outputsfortype.append(output->GetUniqueId());
100 
101  if (!outputsfortype.isEmpty())
102  result.insert(TorcOutput::TypeToString(static_cast<TorcOutput::Type>(type)), outputsfortype);
103  }
104  return result;
105 }

115 void TorcOutputs::AddOutput(TorcOutput *Output)
116 {
117  QWriteLocker locker(&m_httpServiceLock);
118  if (!Output)
119  return;
120 
121  if (Output->GetUniqueId().isEmpty())
122  return;
123 
124  if (outputList.contains(Output))
125  {
126  LOG(VB_GENERAL, LOG_WARNING, QString("Already have output named %1 - ignoring").arg(Output->GetUniqueId()));
127  return;
128  }
129 
130  Output->UpRef();
131  outputList.append(Output);
132  LOG(VB_GENERAL, LOG_INFO, QString("Registered output %1").arg(Output->GetUniqueId()));
133  emit OutputsChanged();
134 }
