103 bool TorcUser::SetUserCredentials(const QString &Name, const QString &Credentials)
104 {
105  QMutexLocker locker(&m_lock);
106 
107  if (Name == gUserName && Credentials.toLower() == GetCredentials())
108  {
109  LOG(VB_GENERAL, LOG_WARNING, "New credentials match old - not changing");
110  return false;
111  }
112 
113  // enforce minimum size (4) and alphanumeric and underscore
114  static QRegExp gReg1("[\\w]{4,}");
115  if (!gReg1.exactMatch(Name))
116  {
117  LOG(VB_GENERAL, LOG_WARNING, "Password unacceptable");
118  return false;
119  }
120 
121  // N.B. MD5 credentials are a hexadecimal representation of a binary. As such they can
122  // contain either upper or lower case digits.
123  static QRegExp gReg2("[0-9a-fA-F]{32}");
124  if (!gReg2.exactMatch(Credentials))
125  {
126  LOG(VB_GENERAL, LOG_WARNING, "User credentials hash unacceptable");
127  return false;
128  }
129 
130  // but an MD5 hash with upper case letters produces different results when
131  // rehashed... so set to lower
132  m_userNameSetting->SetValue(QString(Name));
133  m_userCredentials->SetValue(QString(Credentials.toLower()));
134 
135  TorcLocalContext::NotifyEvent(Torc::UserChanged);
136  return true;
137 }
