88  setObjectName(DBName);
89

92 
93  QVariant::Type vtype = defaultValue.type();
94 
95  if (vtype == QVariant::Int && type == Integer)
96  {
97  value = (roles & Persistent) ? gLocalContext->GetSetting(m_dbName, (int)defaultValue.toInt()) : defaultValue.toInt();
98  }
99  else if (vtype == QVariant::Bool && type == Bool)
100  {
101  value = (roles & Persistent) ? gLocalContext->GetSetting(m_dbName, (bool)defaultValue.toBool()) : defaultValue.toBool();
102  }
103  else if (vtype == QVariant::String)
104  {
105  value = (roles & Persistent) ? gLocalContext->GetSetting(m_dbName, (QString)defaultValue.toString()) : defaultValue.toString();
106  }
107  else if (vtype == QVariant::StringList)
108  {
109  if (roles & Persistent)
110  {
111  QString svalue = gLocalContext->GetSetting(m_dbName, (QString)defaultValue.toString());
112  value = QVariant(svalue.split(","));
113  }
114  else
115  {
116  value = defaultValue.toStringList();
117  }
118  }
119  else
120  {
121  if (vtype != QVariant::Invalid)
122  LOG(VB_GENERAL, LOG_ERR, QString("Unsupported setting data type for %1 (%2)").arg(m_dbName).arg(vtype));
123  value = QVariant();
124  }
125 }

325 bool TorcSetting::SetValue(const QVariant &Value)
326 {
327  QWriteLocker locker(&m_lock);
328  if (value == Value)
329  return true;
330 
331  QVariant::Type vtype = defaultValue.type();
332  if (vtype == QVariant::Int)
333  {
334  int ivalue = Value.toInt();
335  bool valid = false;
336  if (!selections.isEmpty())
337  {
338  valid = selections.contains(QString::number(ivalue));
339  }
340  else if (ivalue >= m_begin && ivalue <= m_end)
341  {
342  valid = true;
343  }
344 
345  if (valid)
346  {
347  if (roles & Persistent)
348  gLocalContext->SetSetting(m_dbName, (int)ivalue);
349  value = Value;
350  locker.unlock();
351  emit ValueChanged(ivalue);
352  return true;
353  }
354  }
355  else if (vtype == QVariant::Bool)
356  {
357  bool bvalue = Value.toBool();
358  if (roles & Persistent)
359  gLocalContext->SetSetting(m_dbName, (bool)bvalue);
360  value = Value;
361  locker.unlock();
362  emit ValueChanged(bvalue);
363  return true;
364  }
365  else if (vtype == QVariant::String)
366  {
367  QString svalue = Value.toString();
368  if (selections.isEmpty() ? true : selections.contains(svalue))
369  {
370  if (roles & Persistent)
371  gLocalContext->SetSetting(m_dbName, svalue);
372  value = Value;
373  locker.unlock();
374  emit ValueChanged(svalue);
375  return true;
376  }
377  }
378  else if (vtype == QVariant::StringList)
379  {
380  QStringList svalue = Value.toStringList();
381  if (roles & Persistent)
382  gLocalContext->SetSetting(m_dbName, svalue.join(","));
383  value = Value;
384  locker.unlock();
385  emit ValueChanged(svalue);
386  return true;
387  }
388  return false;
389 }
