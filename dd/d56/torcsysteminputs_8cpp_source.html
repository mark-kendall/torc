37 void TorcSystemInputs::Create(const QVariantMap &Details)
38 {
39  QWriteLocker locker(&m_handlerLock);
40 
41  QVariantMap::const_iterator i = Details.begin();
42  for ( ; i != Details.end(); ++i)
43  {
44  // network devices can be <sensors> or <outputs>
45  if (i.key() != INPUTS_DIRECTORY)
46  continue;
47 
48  QVariantMap devices = i.value().toMap();
49  QVariantMap::const_iterator j = devices.constBegin();
50  for ( ; j != devices.constEnd(); ++j)
51  {
52  // look for <network>
53  if (j.key() != SYSTEM_INPUTS_STRING)
54  continue;
55 
56  QVariantMap details = j.value().toMap();
57  QVariantMap::const_iterator it = details.constBegin();
58  for ( ; it != details.constEnd(); ++it)
59  {
60  // iterate over known types - currently only 'started'
61  if (it.key() == TorcInput::TypeToString(TorcInput::SystemStarted))
62  {
63  QVariantMap system = it.value().toMap();
64  QString uniqueid = system.value("name").toString();
65  TorcSystemInput *input = new TorcSystemInput(0, system);
66  m_inputs.insert(uniqueid, input);
67  }
68  }
69  }
70  }
