145 void TorcCommandLine::AddPriv(const QString Keys, const QVariant &Default, const QString &HelpText, TorcCommandLine::Options Flags /*= TorcCommandLine::None*/, bool ExitImmediately /*=false*/)
146 {
147  QStringList keys = Keys.split(",");
148 
149  QStringList valid;
150 
151  bool first = true;
152  QString master;
153  foreach (QString key, keys)
154  {
155  if (key.contains("="))
156  {
157  std::cout << QString("Invalid option '%1'").arg(key).toLocal8Bit().constData() << std::endl;
158  }
159  else if (first && !m_options.contains(key))
160  {
161  master = key;
162  m_options.insert(key, TorcArgument(Default, HelpText, Flags, ExitImmediately));
163  valid.append("-" + key);
164  first = false;
165  }
166  else if (!first && !m_aliases.contains(key) && !m_options.contains(key))
167  {
168  m_aliases.insert(key, master);
169  valid.append("-" + key);
170  }
171  else
172  {
173  std::cout << QString("Command line option '%1' already in use - ignoring").arg(key).toLocal8Bit().constData() << std::endl;
174  }
175  }
176 
177  if (!valid.isEmpty())
178  {
179  QString options = valid.join(" OR ");
180  m_help.insert(options, HelpText);
181  if (options.size() > (int)m_maxLength)
182  m_maxLength = options.size() + 2;
183  }
184 }

192 int TorcCommandLine::Evaluate(int argc, const char * const *argv, bool &Exit)
193 {
194  QString error;
195  bool parserror = false;
196  int result = TORC_EXIT_OK;
197  bool printhelp = false;
198  bool printversion = false;
199 
200  // loop through the command line arguments
201  for (int i = 1; i < argc; ++i)
202  {
203  QString key = QString::fromLocal8Bit(argv[i]);
204 
205  // remove trailing '-'s
206  while (key.startsWith("-"))
207  key = key.mid(1);
208 
209  QString value("");
210 
211  bool simpleformat = key.contains("=");
212  if (simpleformat)
213  {
214  // option is --key=value format
215  QStringList keyval = key.split("=");
216  key = keyval.at(0).trimmed();
217  value = keyval.at(1).trimmed();
218  }
219 
220  // do we recognise this option
221  if (!m_options.contains(key))
222  {
223  // is it an alias?
224  if (!m_aliases.contains(key))
225  {
226  parserror = true;
227  error = QString("Unknown command line option '%1'").arg(key);
228  break;
229  }
230  else
231  {
232  key = m_aliases.value(key);
233  }
234  }
235 
236  TorcArgument& argument = m_options[key];
237 
238  // --key value format
239  if (!simpleformat && argument.m_value.isValid())
240  {
241  if (i >= argc - 1)
242  {
243  parserror = true;
244  error = QString("Insufficient arguments - option '%1' requires a value").arg(key);
245  break;
246  }
247 
248  value = QString::fromLocal8Bit(argv[++i]).trimmed();
249 
250  if (value.startsWith("-"))
251  {
252  parserror = true;
253  error = QString("Option '%1' expects a value").arg(key);
254  break;
255  }
256  }
257 
258  if (!argument.m_value.isValid())
259  {
260  if (!value.isEmpty())
261  {
262  // unexpected value
263  parserror = true;
264  error = QString("Option '%1' does not expect a value ('%2')").arg(key).arg(value);
265  break;
266  }
267  else
268  {
269  // mark option as detected
270  argument.m_value = QVariant((bool)true);
271  }
272  }
273  else if (argument.m_value.isValid() && value.isEmpty())
274  {
275  parserror = true;
276  error = QString("Option '%1' expects a value").arg(key).toLocal8Bit().constData();
277  break;
278  }
279 
280  Exit |= argument.m_exitImmediately;
281  printhelp |= (bool)(argument.m_flags & TorcCommandLine::Help);
282  printversion |= (bool)(argument.m_flags & TorcCommandLine::Version);
283 
284  // update the value for the option
285  if (!value.isEmpty())
286  {
287  switch ((QMetaType::Type)argument.m_value.type())
288  {
289  case QMetaType::QStringList:
290  argument.m_value = QVariant(QStringList(value));
291  break;
292  default:
293  argument.m_value = QVariant(value);
294  }
295  }
296  }
297 
298  if (parserror)
299  {
300  result = TORC_EXIT_INVALID_CMDLINE;
301  Exit = true;
302  printhelp = true;
303  std::cout << error.toLocal8Bit().constData() << std::endl << std::endl;
304  }
305 
306  if (printhelp)
307  {
308  std::cout << "Command line options:" << std::endl << std::endl;
309 
310  QHash<QString,QString>::iterator it = m_help.begin();
311  for ( ; it != m_help.end(); ++it)
312  {
313  QByteArray option(it.key().toLocal8Bit().constData());
314  QByteArray padding(m_maxLength - option.size(), 32);
315  std::cout << option.constData() << padding.constData() << it.value().toLocal8Bit().constData() << std::endl;
316  }
317 
318  std::cout << std::endl << "All options may be preceeded by '-' or '--'" << std::endl;
319 
320  {
321  TorcEnvironmentVariable* variable = TorcEnvironmentVariable::gEnvironmentVariable;
322  if (variable)
323  {
324  std::cout << std::endl << "The following environment variables may be useful:" << std::endl;
325 
326  while (variable)
327  {
328  QByteArray var(variable->m_variable.toLocal8Bit().constData());
329  QByteArray padding(m_maxLength - var.size(), 32);
330  std::cout << var.constData() << padding.constData() << variable->m_description.toLocal8Bit().constData() << std::endl;
331  variable = variable->nextEnvironmentVariable;
332  }
333  }
334  }
335  }
336 
337  if (printversion)
338  {
339  std::cout << "QT Version : " << QT_VERSION_STR << std::endl;
340  }
341 
342  return result;
343 }
