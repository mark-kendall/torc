44 TorcCameraParams::TorcCameraParams(const QVariantMap &Details)
45  : m_valid(false),
46  m_width(0),
47  m_height(0),
48  m_stride(0),
49  m_sliceHeight(0),
50  m_frameRate(1),
51  m_bitrate(0),
52  m_timebase(VIDEO_TIMEBASE),
53  m_segmentLength(0),
54  m_gopSize(0),
55  m_videoCodec(),
56  m_contentDir()
57 {
58  if (!Details.contains("width") || !Details.contains("height"))
59  return;
60 
61  bool bitrate = Details.contains("bitrate");
62  bool framerate = Details.contains("framerate");
63  bool video = bitrate && framerate;
64  bool stills = !bitrate && !framerate;
65 
66  if (!video && !stills)
67  return;
68 
69  m_width = Details.value("width").toInt();
70  m_height = Details.value("height").toInt();
71 
72  // N.B. pitch and slice height may be further constrained in certain encoders and may be further adjusted
73  // most h264 streams will expect 16 pixel aligned video so round up
74  int pitch = (m_width + 15) & ~15;
75  if (pitch != m_width)
76  {
77  LOG(VB_GENERAL, LOG_INFO, QString("Rounding video width up to %1 from %2").arg(pitch).arg(m_width));
78  m_width = pitch;
79  }
80 
81  // and 8 pixel macroblock height
82  int height = (m_height + 7) & ~7;
83  if (height != m_height)
84  {
85  LOG(VB_GENERAL, LOG_INFO, QString("Rounding video height up to %1 from %2").arg(height).arg(m_height));
86  m_height = height;
87  }
88 
89  m_stride = m_width;
90  m_sliceHeight = m_height;
91 
92  bool forcemin = m_width < VIDEO_WIDTH_MIN || m_height < VIDEO_HEIGHT_MIN;
93  bool forcemax = m_width > VIDEO_WIDTH_MAX || m_height > VIDEO_HEIGHT_MAX;
94  if (forcemin)
95  {
96  m_width = VIDEO_WIDTH_MIN;
97  m_height = VIDEO_HEIGHT_MIN;
98  LOG(VB_GENERAL, LOG_WARNING, QString("Video too small - forcing output to %1x%2").arg(m_width).arg(m_height));
99  }
100  else if (forcemax)
101  {
102  m_width = VIDEO_WIDTH_MAX;
103  m_height = VIDEO_HEIGHT_MAX;
104  LOG(VB_GENERAL, LOG_WARNING, QString("Video too large - forcing output to %1x%2").arg(m_width).arg(m_height));
105  }
106 
107  if (video)
108  {
109  m_frameRate = Details.value("framerate").toInt();
110  m_bitrate = Details.value("bitrate").toInt();
111 
112  if (m_frameRate < VIDEO_FRAMERATE_MIN)
113  {
114  m_frameRate = VIDEO_FRAMERATE_MIN;
115  LOG(VB_GENERAL, LOG_WARNING, QString("Video framerate too low - forcing to %1").arg(m_frameRate));
116  }
117  else if (m_frameRate > VIDEO_FRAMERATE_MAX)
118  {
119  m_frameRate = VIDEO_FRAMERATE_MAX;
120  LOG(VB_GENERAL, LOG_WARNING, QString("Video framerate too high - forcing to %1").arg(m_frameRate));
121  }
122 
123  if (m_bitrate < VIDEO_BITRATE_MIN)
124  {
125  m_bitrate = VIDEO_BITRATE_MIN;
126  LOG(VB_GENERAL, LOG_WARNING, QString("Video bitrate too low - forcing to %1").arg(m_bitrate));
127  }
128  else if (m_bitrate > VIDEO_BITRATE_MAX)
129  {
130  m_bitrate = VIDEO_BITRATE_MAX;
131  LOG(VB_GENERAL, LOG_WARNING, QString("Video bitrate too high - forcing to %1").arg(m_bitrate));
132  }
133 
134  m_segmentLength = m_frameRate * VIDEO_SEGMENT_TARGET;
135  m_gopSize = m_frameRate * VIDEO_GOPDURA_TARGET;
136 
137  LOG(VB_GENERAL, LOG_INFO, QString("Segment length: %1frames %2seconds").arg(m_segmentLength).arg(m_segmentLength / m_frameRate));
138  LOG(VB_GENERAL, LOG_INFO, QString("GOP length: %1frames %2seconds").arg(m_gopSize).arg(m_gopSize / m_frameRate));
139  LOG(VB_GENERAL, LOG_INFO, QString("Camera video : %1x%2@%3fps bitrate %4").arg(m_width).arg(m_height).arg(m_frameRate).arg(m_bitrate));
140  }
141  else
142  {
143  LOG(VB_GENERAL, LOG_INFO, QString("Camera stills: %1x%2").arg(m_width).arg(m_height));
144  }
145 
146  m_valid = true;
147 }

202 TorcCameraParams TorcCameraParams::Combine(const TorcCameraParams &Add)
203 {
204  if (!Add.m_valid)
205  return *this;
206  if (!m_valid)
207  return Add;
208  if (Add == *this)
209  return *this;
210 
211  if (!IsCompatible(Add))
212  {
213  LOG(VB_GENERAL, LOG_ERR, QString("Error combining camera parameters - %1x%2 != %3x%4")
214  .arg(m_width).arg(m_height).arg(Add.m_width).arg(Add.m_height));
215  // this should cover most cases
216  if ((m_width > Add.m_width) || (m_height > Add.m_height))
217  {
218  m_width = Add.m_width;
219  m_height = Add.m_height;
220  }
221  LOG(VB_GENERAL, LOG_WARNING, QString("Using smaller image size %1x%2").arg(m_width).arg(m_height));
222  }
223 
224  // add video data
225  if (IsStill() && Add.IsVideo())
226  {
227  m_bitrate = Add.m_bitrate;
228  m_frameRate = Add.m_frameRate;
229  m_gopSize = Add.m_gopSize;
230  m_stride = Add.m_stride;
231  m_sliceHeight = Add.m_sliceHeight;
232  m_segmentLength = Add.m_segmentLength;
233  m_videoCodec = Add.m_videoCodec;
234  m_timebase = Add.m_timebase;
235  LOG(VB_GENERAL, LOG_INFO, "Added video to camera parameters");
236  }
237  // add stills
238  else if (IsVideo() && Add.IsStill())
239  {
240  m_contentDir = Add.m_contentDir;
241  }
242 
243  return *this;
244 }
