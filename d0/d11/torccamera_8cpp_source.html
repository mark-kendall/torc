47 TorcCameraParams::TorcCameraParams(const QVariantMap &Details)
48  : m_valid(false),
49  m_width(0),
50  m_height(0),
51  m_stride(0),
52  m_sliceHeight(0),
53  m_frameRate(1),
54  m_bitrate(0),
55  m_timebase(VIDEO_TIMEBASE),
56  m_segmentLength(0),
57  m_gopSize(0),
58  m_videoCodec(),
59  m_contentDir()
60 {
61  if (!Details.contains("width") || !Details.contains("height"))
62  return;
63 
64  bool bitrate = Details.contains("bitrate");
65  bool framerate = Details.contains("framerate");
66  bool video = bitrate && framerate;
67  bool stills = !bitrate && !framerate;
68 
69  if (!video && !stills)
70  return;
71 
72  m_width = Details.value("width").toInt();
73  m_height = Details.value("height").toInt();
74 
75  // N.B. pitch and slice height may be further constrained in certain encoders and may be further adjusted
76  // most h264 streams will expect 16 pixel aligned video so round up
77  int pitch = (m_width + 15) & ~15;
78  if (pitch != m_width)
79  {
80  LOG(VB_GENERAL, LOG_INFO, QString("Rounding video width up to %1 from %2").arg(pitch).arg(m_width));
81  m_width = pitch;
82  }
83 
84  // and 8 pixel macroblock height
85  int height = (m_height + 7) & ~7;
86  if (height != m_height)
87  {
88  LOG(VB_GENERAL, LOG_INFO, QString("Rounding video height up to %1 from %2").arg(height).arg(m_height));
89  m_height = height;
90  }
91 
92  m_stride = m_width;
93  m_sliceHeight = m_height;
94 
95  bool forcemin = m_width < VIDEO_WIDTH_MIN || m_height < VIDEO_HEIGHT_MIN;
96  bool forcemax = m_width > VIDEO_WIDTH_MAX || m_height > VIDEO_HEIGHT_MAX;
97  if (forcemin)
98  {
99  m_width = VIDEO_WIDTH_MIN;
100  m_height = VIDEO_HEIGHT_MIN;
101  LOG(VB_GENERAL, LOG_WARNING, QString("Video too small - forcing output to %1x%2").arg(m_width).arg(m_height));
102  }
103  else if (forcemax)
104  {
105  m_width = VIDEO_WIDTH_MAX;
106  m_height = VIDEO_HEIGHT_MAX;
107  LOG(VB_GENERAL, LOG_WARNING, QString("Video too large - forcing output to %1x%2").arg(m_width).arg(m_height));
108  }
109 
110  if (video)
111  {
112  m_frameRate = Details.value("framerate").toInt();
113  m_bitrate = Details.value("bitrate").toInt();
114 
115  if (m_frameRate < VIDEO_FRAMERATE_MIN)
116  {
117  m_frameRate = VIDEO_FRAMERATE_MIN;
118  LOG(VB_GENERAL, LOG_WARNING, QString("Video framerate too low - forcing to %1").arg(m_frameRate));
119  }
120  else if (m_frameRate > VIDEO_FRAMERATE_MAX)
121  {
122  m_frameRate = VIDEO_FRAMERATE_MAX;
123  LOG(VB_GENERAL, LOG_WARNING, QString("Video framerate too high - forcing to %1").arg(m_frameRate));
124  }
125 
126  if (m_bitrate < VIDEO_BITRATE_MIN)
127  {
128  m_bitrate = VIDEO_BITRATE_MIN;
129  LOG(VB_GENERAL, LOG_WARNING, QString("Video bitrate too low - forcing to %1").arg(m_bitrate));
130  }
131  else if (m_bitrate > VIDEO_BITRATE_MAX)
132  {
133  m_bitrate = VIDEO_BITRATE_MAX;
134  LOG(VB_GENERAL, LOG_WARNING, QString("Video bitrate too high - forcing to %1").arg(m_bitrate));
135  }
136 
137  m_segmentLength = m_frameRate * VIDEO_SEGMENT_TARGET;
138  m_gopSize = m_frameRate * VIDEO_GOPDURA_TARGET;
139 
140  LOG(VB_GENERAL, LOG_INFO, QString("Segment length: %1frames %2seconds").arg(m_segmentLength).arg(m_segmentLength / m_frameRate));
141  LOG(VB_GENERAL, LOG_INFO, QString("GOP length: %1frames %2seconds").arg(m_gopSize).arg(m_gopSize / m_frameRate));
142  LOG(VB_GENERAL, LOG_INFO, QString("Camera video : %1x%2@%3fps bitrate %4").arg(m_width).arg(m_height).arg(m_frameRate).arg(m_bitrate));
143  }
144  else
145  {
146  LOG(VB_GENERAL, LOG_INFO, QString("Camera stills: %1x%2").arg(m_width).arg(m_height));
147  }
148 
149  m_valid = true;
150 }

189 TorcCameraParams TorcCameraParams::Combine(const TorcCameraParams &Add)
190 {
191  if (!Add.m_valid)
192  return *this;
193  if (!m_valid)
194  return Add;
195  if (Add == *this)
196  return *this;
197 
198  if (!IsCompatible(Add))
199  {
200  LOG(VB_GENERAL, LOG_ERR, QString("Error combining camera parameters - %1x%2 != %3x%4")
201  .arg(m_width).arg(m_height).arg(Add.m_width).arg(Add.m_height));
202  // this should cover most cases
203  if ((m_width > Add.m_width) || (m_height > Add.m_height))
204  {
205  m_width = Add.m_width;
206  m_height = Add.m_height;
207  }
208  LOG(VB_GENERAL, LOG_WARNING, QString("Using smaller image size %1x%2").arg(m_width).arg(m_height));
209  }
210 
211  // add video data
212  if (IsStill() && Add.IsVideo())
213  {
214  m_bitrate = Add.m_bitrate;
215  m_frameRate = Add.m_frameRate;
216  m_gopSize = Add.m_gopSize;
217  m_stride = Add.m_stride;
218  m_sliceHeight = Add.m_sliceHeight;
219  m_segmentLength = Add.m_segmentLength;
220  m_videoCodec = Add.m_videoCodec;
221  m_timebase = Add.m_timebase;
222  LOG(VB_GENERAL, LOG_INFO, "Added video to camera parameters");
223  }
224  // add stills
225  else if (IsVideo() && Add.IsStill())
226  {
227  m_contentDir = Add.m_contentDir;
228  }
229 
230  return *this;
231 }

285 bool TorcCameraDevice::Setup(void)
286 {
287  QWriteLocker locker(&m_ringBufferLock);
288  int buffersize = (m_params.m_bitrate * (m_params.m_segmentLength / m_params.m_frameRate) * VIDEO_SEGMENT_NUMBER) / 8;
289  m_ringBuffer = new TorcSegmentedRingBuffer(buffersize, VIDEO_SEGMENT_MAX);
290  m_muxer = new TorcMuxer(m_ringBuffer);
291  if (!m_muxer)
292  return false;
293 
294  m_videoStream = m_muxer->AddH264Stream(m_params.m_width, m_params.m_height, VIDEO_H264_PROFILE, m_params.m_bitrate);
295  if (!m_muxer->IsValid())
296  return false;
297 
298  connect(m_ringBuffer, &TorcSegmentedRingBuffer::SegmentReady, this, &TorcCameraDevice::SegmentReady);
299  connect(m_ringBuffer, &TorcSegmentedRingBuffer::SegmentRemoved, this, &TorcCameraDevice::SegmentRemoved);
300  connect(m_ringBuffer, &TorcSegmentedRingBuffer::InitSegmentReady, this, &TorcCameraDevice::InitSegmentReady);
301 
302  m_frameCount = 0;
303  m_bufferedPacket = nullptr;
304  m_haveInitSegment = false;
305  return true;
306 }

347 void TorcCameraDevice::SaveStill(void)
348 {
349  if (m_stillsExpected)
350  {
351  if (m_stillsRequired)
352  {
353  if (!m_params.m_contentDir.isEmpty())
354  {
355  QFile file(m_params.m_contentDir + QDateTime::currentDateTime().toString("yyyy_MM_dd_hh_mm_ss_zzz") + ".jpg");
356  if (file.open(QIODevice::ReadWrite | QIODevice::Truncate))
357  {
358  QPair<quint32, uint8_t*> pair;
359  foreach(pair, m_stillsBuffers)
360  file.write((const char*)pair.second, pair.first);
361  file.close();
362  LOG(VB_GENERAL, LOG_INFO, QString("Saved snapshot as '%1'").arg(file.fileName()));
363  emit StillReady(file.fileName());
364  }
365  else
366  {
367  LOG(VB_GENERAL, LOG_ERR, QString("Failed to open %1 for writing").arg(file.fileName()));
368  }
369  }
370  m_stillsRequired--;
371  }
372  m_stillsExpected--;
373  }
374  ClearStillsBuffers();
375 }

393 void TorcCameraDevice::TrackDrift(void)
394 {
395  quint64 now = QDateTime::currentMSecsSinceEpoch();
396  if (!m_referenceTime)
397  m_referenceTime = now;
398 
399  // discard first few samples while camera settles down
400  if (m_discardDrift)
401  {
402  m_discardDrift--;
403  return;
404  }
405 
406  // reference time points to the end of the first video segment
407  m_ringBufferLock.lockForRead();
408  int last = m_ringBuffer->GetHead();
409  m_ringBufferLock.unlock();
410  if (last < 0)
411  return;
412 
413  // all milliseconds
414  qint64 drift = QDateTime::currentMSecsSinceEpoch() - (m_referenceTime + (last * VIDEO_SEGMENT_TARGET * 1000));
415  double shortaverage = m_shortAverage.AddValue(drift);
416  double longaverage = m_longAverage.AddValue(drift);
417 
418  static const double timedelta = ((VIDEO_DRIFT_LONG - VIDEO_DRIFT_SHORT) / 2) * 1000;
419  double driftdelta = longaverage - shortaverage;
420  double gradient = driftdelta / timedelta;
421  double timetozero = qAbs(driftdelta) < 1 ? qInf() : drift / gradient;
422 
423  LOG(VB_GENERAL, LOG_INFO, QString("Drift: %1secs 1min %2 5min %3 timetozero %4")
424  .arg((double)drift/1000, 0, 'f', 3).arg(shortaverage/1000, 0, 'f', 3).arg(longaverage/1000, 0, 'f', 3).arg(timetozero/1000, 0, 'f', 3));
425 }
