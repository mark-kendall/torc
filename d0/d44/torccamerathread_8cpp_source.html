37 void TorcCameraThread::CreateOrDestroy(TorcCameraThread *&Thread, const QString &Type, const TorcCameraParams &Params)
38 {
39  static QMutex lock(QMutex::NonRecursive);
40  static QMutexLocker locker(&lock);
41  static QMap<QString,TorcCameraThread*> threads;
42 
43  if (Thread)
44  {
45  if (threads.contains(Type))
46  {
47  TorcCameraThread* thread = threads.value(Type);
48  if (thread != Thread)
49  LOG(VB_GENERAL, LOG_ERR, QString("Mismatch between thread and type for '%1'").arg(Type));
50 
51  thread->DownRef();
52  // release our own ref
53  if (!thread->IsShared())
54  {
55  LOG(VB_GENERAL, LOG_INFO, QString("Removing camera thread - no longer in use '%1'").arg(Type));
56  thread->DownRef();
57  threads.remove(Type);
58  }
59  }
60  else
61  {
62  LOG(VB_GENERAL, LOG_ERR, QString("Unknown camera thread type '%1'").arg(Type));
63  }
64  }
65  else
66  {
67  TorcCameraThread* thread = NULL;
68  if (threads.contains(Type))
69  {
70  LOG(VB_GENERAL, LOG_INFO, QString("Sharing existing camera thread '%1'").arg(Type));
71  thread = threads.value(Type);
72  }
73  else
74  {
75  LOG(VB_GENERAL, LOG_INFO, QString("Creating shared camera thread '%1'").arg(Type));
76  thread = new TorcCameraThread(Type, Params);
77  threads.insert(Type, thread);
78  }
79  // as creator, we hold one reference. Add another for the requestor.
80  thread->UpRef();
81  Thread = thread;
82  }
83 }
